75 typedef struct {

79  apiflt ambient; /* ambient lighting */

82  apiflt opacity; /* how opaque the object is */
83  vector ctr; /* origin of texture */
84  vector rot; /* rotation of texture around origin */
85  vector scale; /* scale of texture in x,y,z */
86  vector uaxs; /* planar map u axis */
87  vector vaxs; /* planar map v axis */
88  char imap[96]; /* name of image map */
89 } apitexture;

92 /*******************************************************************
93  * NOTE: The value passed in apitexture.texturefunc corresponds to
94  * the meanings given in this table:
95  *
96  * 0 - No texture function is applied other than standard lighting.
97  * 1 - 3D checkerboard texture. Red & Blue checkers through 3d space.
98  * 2 - Grit texture, roughens up the surface of the object a bit.
99  * 3 - 3D marble texture. Makes a 3D swirl pattern through the object.
100  * 4 - 3D wood texture. Makes a 3D wood pattern through the object.
101  * 5 - 3D gradient noise function.
102  * 6 - I've forgotten :-)
103  * 7 - Cylindrical Image Map **** IMAGE MAPS REQUIRE the filename
104  * 8 - Spherical Image Map of the image be put in imap[]
105  * 9 - Planar Image Map part of the texture...
106  * planar requires uaxs, and vaxs..
107  *
108  *******************************************************************/
