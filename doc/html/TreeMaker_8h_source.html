40  public:
41  SubTreeCreationTask( TreeNode*& root, long number_of_nodes ) : my_root(root), is_continuation(false) {
42  my_root = MyTreeMaker::allocate_node();
43  my_root->node_count = number_of_nodes;
44  my_root->value = Value(Pi*number_of_nodes);
45  }
46 
47  tbb::task* execute() {
48  tbb::task* next = NULL;
49  if( !is_continuation ) {
50  long subtree_size = my_root->node_count - 1;
51  if( subtree_size<1000 ) { /* grainsize */
52  my_root->left = MyTreeMaker::do_in_one_thread(subtree_size/2);
53  my_root->right = MyTreeMaker::do_in_one_thread(subtree_size - subtree_size/2);
54  } else {
55  // Create tasks before spawning any of them.
56  tbb::task* a = new( allocate_child() ) SubTreeCreationTask(my_root->left,subtree_size/2);
57  tbb::task* b = new( allocate_child() ) SubTreeCreationTask(my_root->right,subtree_size - subtree_size/2);
58  recycle_as_continuation();
59  is_continuation = true;
60  set_ref_count(2);
61  spawn(*b);
62  next = a;
63  }
64  }
65  return next;
66  }
67  };

74  static TreeNode* do_in_one_thread( long number_of_nodes ) {
75  if( number_of_nodes==0 ) {
76  return NULL;
77  } else {
78  TreeNode* n = allocate_node();
79  n->node_count = number_of_nodes;
80  n->value = Value(Pi*number_of_nodes);
81  --number_of_nodes;
82  n->left = do_in_one_thread( number_of_nodes/2 );
83  n->right = do_in_one_thread( number_of_nodes - number_of_nodes/2 );
84  return n;
85  }
86  }

95  static TreeNode* create_and_time( long number_of_nodes, bool silent=false ) {
96  tbb::tick_count t0, t1;
97  TreeNode* root = allocate_node();
98  root->node_count = number_of_nodes;
99  root->value = Value(Pi*number_of_nodes);
100  --number_of_nodes;
101 
102  t0 = tbb::tick_count::now();
103  root->left = do_in_one_thread( number_of_nodes/2 );
104  t1 = tbb::tick_count::now();
105  if ( !silent ) printf ("%24s: time = %.1f msec\n", "half created serially", (t1-t0).seconds()*1000);
106 
107  t0 = tbb::tick_count::now();
108  root->right = do_in_parallel( number_of_nodes - number_of_nodes/2 );
109  t1 = tbb::tick_count::now();
110  if ( !silent ) printf ("%24s: time = %.1f msec\n", "half done in parallel", (t1-t0).seconds()*1000);
111
