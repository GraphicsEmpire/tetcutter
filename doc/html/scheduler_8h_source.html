587 inline void generic_scheduler::commit_spawned_tasks( size_t new_tail ) {
588  __TBB_ASSERT ( new_tail <= my_arena_slot->my_task_pool_size, "task deque end was overwritten" );

596 void generic_scheduler::commit_relocated_tasks ( size_t new_tail ) {
597  __TBB_ASSERT( is_local_task_pool_quiescent(),
598  "Task pool must be locked when calling commit_relocated_tasks()" );
599  __TBB_store_relaxed( my_arena_slot->head, 0 );
600  // Tail is updated last to minimize probability of a thread making arena
601  // snapshot being misguided into thinking that this task pool is empty.

603  release_task_pool();
604 }

606 template<free_task_hint hint>
607 void generic_scheduler::free_task( task& t ) {
608 #if __TBB_HOARD_NONLOCAL_TASKS
609  static const int h = hint&(~local_task);
610 #else
611  static const free_task_hint h = hint;
612 #endif
613  GATHER_STATISTIC(--my_counters.active_tasks);
614  task_prefix& p = t.prefix();
615  // Verify that optimization hints are correct.
616  __TBB_ASSERT( h!=small_local_task || p.origin==this, NULL );
617  __TBB_ASSERT( !(h&small_task) || p.origin, NULL );
618  __TBB_ASSERT( !(h&local_task) || (!p.origin || uintptr_t(p.origin) > uintptr_t(4096)), "local_task means allocated");
619  poison_value(p.depth);
620  poison_value(p.ref_count);
621  poison_pointer(p.owner);
622  __TBB_ASSERT( 1L<<t.state() & (1L<<task::executing|1L<<task::allocated), NULL );
623  p.state = task::freed;
624  if( h==small_local_task || p.origin==this ) {
625  GATHER_STATISTIC(++my_counters.free_list_length);
626  p.next = my_free_list;
627  my_free_list = &t;
628  } else if( !(h&local_task) && p.origin && uintptr_t(p.origin) < uintptr_t(4096) ) {
629  // a special value reserved for future use, do nothing since
630  // origin is not pointing to a scheduler instance
631  } else if( !(h&local_task) && p.origin ) {
632  GATHER_STATISTIC(++my_counters.free_list_length);
633 #if __TBB_HOARD_NONLOCAL_TASKS
634  if( !(h&no_cache) ) {
635  p.next = my_nonlocal_free_list;
636  my_nonlocal_free_list = &t;
637  } else
638 #endif
639  free_nonlocal_small_task(t);
640  } else {
641  GATHER_STATISTIC(--my_counters.big_tasks);
642  deallocate_task(t);
643  }
644 }

646 #if __TBB_TASK_PRIORITY
647 inline intptr_t generic_scheduler::effective_reference_priority () const {
648  // Workers on the outermost dispatch level (i.e. with empty stack) use market's
649  // priority as a reference point (to speedup discovering process level priority
650  // changes). But when there are enough workers to service (even if only partially)
651  // a lower priority arena, they should use arena's priority as a reference, lest
652  // be trapped in a futile spinning (because market's priority would prohibit
653  // executing ANY tasks in this arena).
654  return !worker_outermost_level() ||
655  my_arena->my_num_workers_allotted < my_arena->num_workers_active()
656  ? *my_ref_top_priority : my_arena->my_top_priority;
657 }
