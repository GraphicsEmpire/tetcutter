326  void allocate_task_pool( size_t n ) {
327  size_t byte_size = ((n * sizeof(task*) + NFS_MaxLineSize - 1) / NFS_MaxLineSize) * NFS_MaxLineSize;
328  my_task_pool_size = byte_size / sizeof(task*);
329  task_pool_ptr = (task**)NFS_Allocate( 1, byte_size, NULL );

360  // It is possible not to copy memory but just to copy pointers but the following issues should be addressed:

366  cpu_ctl_env( const cpu_ctl_env &src ) : my_fenv_ptr(NULL) {
367  *this = src;

370  __TBB_ASSERT( src.my_fenv_ptr, NULL );
371  if ( !my_fenv_ptr )
372  my_fenv_ptr = (fenv_t*)tbb::internal::NFS_Allocate(1, sizeof(fenv_t), NULL);
373  *my_fenv_ptr = *src.my_fenv_ptr;
374  return *this;
375  }
376  bool operator!=( const cpu_ctl_env &ctl ) const {
377  __TBB_ASSERT( my_fenv_ptr, "cpu_ctl_env is not initialized." );
