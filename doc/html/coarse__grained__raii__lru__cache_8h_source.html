54  handle_object operator[](key_type k){
55  tbb::spin_mutex::scoped_lock lock(my_mutex);
56  bool is_new_value_needed = false;
57  typename map_storage_type::iterator it = my_map_storage.find(k);
58  if (it == my_map_storage.end()){
59  it = my_map_storage.insert(it,std::make_pair(k,map_value_type(value_type(),0,my_lru_list.end(),false)));
60  is_new_value_needed = true;
61  }else {
62  typename lru_list_type::iterator list_it = it->second.my_lru_list_iterator;
63  if (list_it!=my_lru_list.end()) {
64  my_lru_list.erase(list_it);
65  it->second.my_lru_list_iterator= my_lru_list.end();
66  }
67  }
68  typename map_storage_type::reference value_ref = *it;
69  //increase ref count
70  ++(value_ref.second.my_ref_counter);
71  if (is_new_value_needed){
72  lock.release();
73  value_ref.second.my_value = my_value_function(k);
74  __TBB_store_with_release(value_ref.second.my_is_ready, true);
75 
76  }else{
77  if (!value_ref.second.my_is_ready){
78  lock.release();
79  tbb::internal::spin_wait_while_eq(value_ref.second.my_is_ready,false);
80  }
81  }
82  return handle_object(*this,(value_ref));
83  }

85  void signal_end_of_usage(typename map_storage_type::reference value_ref){
86  tbb::spin_mutex::scoped_lock lock(my_mutex);
87  typename map_storage_type::iterator it = my_map_storage.find(value_ref.first);
88  __TBB_ASSERT(it!=my_map_storage.end(),"cache should not return past-end iterators to outer world");
89  __TBB_ASSERT(&(*it) == &value_ref,"dangling reference has been returned to outside world? data race ?");
90  __TBB_ASSERT( my_lru_list.end()== std::find(my_lru_list.begin(),my_lru_list.end(),it),
91  "object in use should not be in list of unused objects ");
92  if (! --(it->second.my_ref_counter)){ //decrease ref count, and check if it was the last reference
93  if (my_lru_list.size()>=my_number_of_lru_history_items){
94  size_t number_of_elements_to_evict = 1 + my_lru_list.size() - my_number_of_lru_history_items;
95  for (size_t i=0; i<number_of_elements_to_evict; ++i){
96  typename map_storage_type::iterator it_to_evict = my_lru_list.back();
97  my_lru_list.pop_back();
98  my_map_storage.erase(it_to_evict);
99  }
100  }
101  my_lru_list.push_front(it);
102  it->second.my_lru_list_iterator = my_lru_list.begin();
103  }
104  }
