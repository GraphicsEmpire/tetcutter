91 void my_affinity_partitioner_base_v3::resize( unsigned factor ) {
92  // Check factor to avoid asking for number of workers while there might be no arena.
93  size_t new_size = factor ? factor * g_threadNums.local() : 0;
94  if (new_size != my_size) {
95  if (my_array) {
96  NFS_Free(my_array);
97  // Following two assignments must be done here for sake of exception safety.

104  my_size = new_size;
105  }

116 class start_for : public fake_task {
117  Range my_range;
118  Body my_body;
119  typename Partitioner::task_partition_type my_partition;

124 public:
125  start_for( const Range& range, const Body& body, Partitioner& partitioner,
126  test_partitioner_utils::BinaryTree* tree ) :
127  my_range(range), my_body(body), my_partition(partitioner),
128  m_executedBegin(0), m_executedEnd(0), m_firstTimeRun(true),
129  m_joinedBegin(/* grows left */ range.end()), m_joinedEnd(range.end()), m_tree(tree)
130  {
131  if (m_tree) {
132  m_tree->push_node( test_partitioner_utils::make_node(my_range.begin(), my_range.end(), affinity()) );

137  start_for( start_for& parent_, typename Partitioner::split_type& split_obj) :
138  my_range(parent_.my_range, split_obj),
139  my_body(parent_.my_body),
140  my_partition(parent_.my_partition, split_obj),
141  m_executedBegin(0), m_executedEnd(0), m_firstTimeRun(true),
142  m_joinedBegin(/* grows left */ my_range.end()), m_joinedEnd(my_range.end()),
143  m_tree(parent_.m_tree)
144  {
145  set_parent(parent_.parent());
146  my_partition.set_affinity(*this);
147 
148  if (m_tree) {
149  // collecting splitting statistics
150  m_tree->push_node( test_partitioner_utils::make_node(my_range.begin(),
151  my_range.end(),
152  affinity()) );
153  m_tree->push_node( test_partitioner_utils::make_node(parent_.my_range.begin(),
154  parent_.my_range.end(),
155  parent_.affinity()) );
156  }
157  }

160  start_for( start_for& parent_, const Range& r, depth_t d ) :
161  my_range(r),
162  my_body(parent_.my_body),
163  my_partition(parent_.my_partition, tbb::split()),
164  m_executedBegin(0), m_executedEnd(0), m_firstTimeRun(true),
165  m_joinedBegin(/* grows left */ r.end()), m_joinedEnd(r.end()),
166  m_tree(parent_.m_tree)

168  set_parent(parent_.parent());
169  my_partition.set_affinity(*this);
170  my_partition.align_depth( d );
171  }
172  fake_task* execute() {
173  my_partition.check_being_stolen( *this );
174  size_t origBegin = my_range.begin();
175  size_t origEnd = my_range.end();
176 
177  my_partition.execute(*this, my_range);
178 
179  ASSERT(m_executedEnd == m_joinedBegin, "Non-continuous execution");
180  m_executedEnd = m_joinedEnd;
181 
182  ASSERT(origBegin == m_executedBegin && origEnd == m_executedEnd,
183  "Not all iterations were processed");
184  return NULL;
185  }
187  void run_body( Range &r ) {
188  if( r.is_ensure_non_emptiness() )
189  ASSERT( !r.empty(), "Empty ranges are not allowed" );
190  my_body(r);
191  if (m_firstTimeRun) {
192  m_firstTimeRun = false;
193  m_executedBegin = m_executedEnd = r.begin();
194  }
195  ASSERT(m_executedBegin <= r.begin() && m_executedEnd <= r.end(),
196  "Non-continuous execution");
197  m_executedEnd = r.end();
198  }
200  void offer_work(typename Partitioner::split_type& split_obj) {

305  static void uniform_distribution_checker(const char* rangeName, size_t rangeSize, const use_case_settings_t* settings,
306  const RangeStatisticData& stat)
307  {
308  // Checking that all threads were given a task
309  if (rangeSize >= settings->thread_num) {
310  uint64_t disparity =
311  max(stat.m_rangeNum, settings->thread_num) - min(stat.m_rangeNum, settings->thread_num);
312  if (disparity > settings->above_threads_size_tolerance) {
313  REPORT("ERROR: '%s (f=%d|e=%d)': |#ranges-#threads|=%llu > %llu=tolerance\n",
314  rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
315  disparity, uint64_t(settings->above_threads_size_tolerance));
316  ASSERT(disparity <= settings->above_threads_size_tolerance, "Incorrect number of range "
317  "objects was created before work balancing phase started");
318  }
319  } else if (settings->ensure_non_empty_size && rangeSize != 0) {
320  uint64_t disparity = max(stat.m_rangeNum, rangeSize) - min(stat.m_rangeNum, rangeSize);
321  if (disparity > settings->below_threads_size_tolerance ) {
322  REPORT("ERROR: '%s (f=%d|e=%d)': |#ranges-range size|=%llu > %llu=tolerance\n",
323  rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
324  disparity, uint64_t(settings->below_threads_size_tolerance));
325  ASSERT(disparity <= settings->below_threads_size_tolerance, "Incorrect number of range objects"
326  " was created before work balancing phase started");
327  }
328  }
329  // Checking difference between min and max number of range iterations
330  size_t diff = stat.m_maxRangeSize - stat.m_minRangeSize;
331  if (diff > settings->between_min_max_ranges_tolerance) {
332  REPORT("ERROR: '%s (f=%d|e=%d)': range size difference=%llu > %llu=tolerance\n",
333  rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
334  uint64_t(diff), uint64_t(settings->between_min_max_ranges_tolerance));
335  ASSERT(diff <= settings->between_min_max_ranges_tolerance, "Uniform iteration distribution error");
336  }
337  }
338  // Checker for test cases where ranges don't provide feedback during proportional split to
339  // partitioner and differ from tbb::blocked_range implementation in their splitting algorithm
340  static void nonuniform_distribution_checker(const char* rangeName, size_t rangeSize, const use_case_settings_t* settings,
341  const RangeStatisticData& stat)
342  {
343  if (stat.m_rangeNum > settings->thread_num) {
344  REPORT("ERROR: '%s (f=%d|e=%d)': %llu=#ranges > #threads=%llu\n",
345  rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
346  uint64_t(stat.m_rangeNum), uint64_t(settings->thread_num));
347  ASSERT(stat.m_rangeNum <= settings->thread_num,
348  "Incorrect number of range objects was created before work balancing phase started");
349  }
350  // Checking difference between min and max number of range iterations
351  size_t diff = stat.m_maxRangeSize - stat.m_minRangeSize;
352  if (diff > rangeSize) {
353  REPORT("ERROR: '%s (f=%d|e=%d)': range size difference=%llu > %llu=initial range size\n",
354  rangeName, int(settings->provide_feedback), int(settings->ensure_non_empty_size),
355  uint64_t(diff), uint64_t(rangeSize));
356  ASSERT(diff <= rangeSize, "Iteration distribution error");
357  }
358  }

363  template <typename Range, typename T>
364  void test(use_case_settings_t& settings, T factors[], size_t (*rsgFunc)(T*, unsigned, size_t)
365  = &default_range_size_generator<T>) const
366  {
367  for (unsigned i = 0; i < settings.factors_array_len; ++i) {
368  size_t range_end = rsgFunc(factors, i, settings.thread_num);
369  RangeStatisticData stat = { /*range num=*/ 0, /*minimal size of range=*/ 0,
370  /*maximal size of range=*/ 0, /*minimal size of range was not rewritten yet=*/ false };
371  Range range = Range(settings.range_begin, range_end, &stat, settings.provide_feedback,
372  settings.ensure_non_empty_size);
373  tbb::affinity_partitioner ap;
374  test_case(range, SimpleBody(), ap, NULL);
375  size_t range_size = range_end - settings.range_begin;
376  const char* rangeName = typeid(range).name();
377  settings.checker(rangeName, range_size, &settings, stat);
378  }
