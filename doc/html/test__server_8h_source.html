69  unallocated,

73  busy,
75  clean
76  };
77  tbb::atomic<int> state;
78  tbb::atomic<int> processing_count;
79  void update( state_t new_state, state_t old_state ) {
80  int o = state.compare_and_swap(new_state,old_state);
81  ASSERT( o==old_state, "illegal transition" );
82  }
83  void update_from_either( state_t new_state, state_t old_state1, state_t old_state2 ) {
84  int snapshot;
85  do {
86  snapshot = state;
87  ASSERT( snapshot==old_state1||snapshot==old_state2, "illegal transition" );
88  } while( state.compare_and_swap(new_state,snapshot)!=snapshot );
89  }
90  MyJob() {
91  state=unallocated;
92  processing_count=0;
93  }
94  ~MyJob() {
95  // Overwrite so that accidental use after destruction can be detected.

266 void Checker::check_number_of_threads_delivered( int n_delivered, int n_requested, int n_extra ) const {
267  ASSERT( default_concurrency>=0, NULL );
268  if( tracker.all_set() ) return;
269  // Check that number of threads delivered is reasonable.
270  int n_avail = default_concurrency;
271  if( n_extra>0 )
272  n_avail-=n_extra;
273  if( n_avail<0 )
274  n_avail=0;
275  if( n_requested>default_concurrency )
276  n_avail += n_requested-default_concurrency;
277  int n_expected = n_requested;
278  if( n_expected>n_avail )
279  n_expected=n_avail;
280  const char* msg = NULL;
281  if( n_delivered>n_avail ) {
282  if( ++tracker.n_more_than_available>1 )
283  return;
284  msg = "server delivered more threads than were theoretically available";
285  } else if( n_delivered>n_expected ) {
286  if( ++tracker.n_too_many_threads>1 )
287  return;
288  msg = "server delivered more threads than expected";
289  } else if( n_delivered<n_expected ) {
290  if( ++tracker.n_system_overload>1 )
291  return;
292  msg = "server delivered fewer threads than ideal; or, the system is overloaded?";
293  }
294  if( msg ) {
295  REPORT("Warning: %s (n_delivered=%d n_avail=%d n_requested=%d n_extra=%d default_concurrency=%d)\n",
296  msg, n_delivered, n_avail, n_requested, n_extra, default_concurrency );
297  }
298 }

323 template<typename Factory,typename Client>
324 void DoOneConnection<Factory,Client>::operator()( size_t i ) const {
325  ASSERT( i<MaxClient, NULL );
326  Client* client = new Client;
327  client->initialize( Client::is_omp ? JobArraySize : n_thread, nesting, ClientStackSize[i] );
328  Factory factory;
329  memset( &factory, 0, sizeof(factory) );
330  typename Factory::status_type status = factory.open();
331  ASSERT( status==Factory::st_success, NULL );
332 
333  typename Factory::server_type* server;
334  status = factory.make_server( server, *client );
335  ASSERT( status==Factory::st_success, NULL );
336  Harness::ConcurrencyTracker ct;
337  REMARK("client %d: opened server n_thread=%d nesting=(%d,%d)\n",
338  client->client_id(), n_thread, nesting.level, nesting.limit);
339  client->set_server( server );
340  Checker checker( *server );
341 
342  FireUpJobs( *server, *client, n_thread, n_extra, check_delivered && !client->is_strict() ? &checker : NULL );
343 
344  // Close the connection
345  client->expect_close_connection = true;
346  REMARK("client %d: calling request_close_connection\n", client->client_id());
347 #if !RML_USE_WCRM
348  int default_concurrency = server->default_concurrency();
349 #endif
350  server->request_close_connection();
351  // Client deletes itself when it sees call to acknowledge_close_connection from server.
352  factory.close();
353 #if !RML_USE_WCRM
354  if( TestSingleConnection )
355  __TBB_ASSERT_EX( uintptr_t(factory.scratch_ptr)==uintptr_t(default_concurrency), "under/over subscription?" );
356 #endif
357 }
