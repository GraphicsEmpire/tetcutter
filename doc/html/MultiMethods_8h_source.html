62 // Implements an automatic static double dispatcher based on two typelists
64 
65  template
66  <
67  class Executor,
68  class BaseLhs,
69  class TypesLhs,
70  bool symmetric = true,
71  class BaseRhs = BaseLhs,
72  class TypesRhs = TypesLhs,
73  typename ResultType = void
74  >
75  class StaticDispatcher
76  {
77  template <class SomeLhs>

79  Executor exec, NullType)

82  template <class Head, class Tail, class SomeLhs>
83  static ResultType DispatchRhs(SomeLhs& lhs, BaseRhs& rhs,
84  Executor exec, Typelist<Head, Tail>)
85  {
86  if (Head* p2 = dynamic_cast<Head*>(&rhs))
87  {
88  Int2Type<(symmetric &&
89  int(TL::IndexOf<TypesRhs, Head>::value) <
90  int(TL::IndexOf<TypesLhs, SomeLhs>::value))> i2t;
91 
92  typedef Private::InvocationTraits<
93  SomeLhs, Head, Executor, ResultType> CallTraits;
94 
95  return CallTraits::DoDispatch(lhs, *p2, exec, i2t);
96  }
97  return DispatchRhs(lhs, rhs, exec, Tail());
98  }

100  static ResultType DispatchLhs(BaseLhs& lhs, BaseRhs& rhs,
101  Executor exec, NullType)
102  { return exec.OnError(lhs, rhs); }
103 
104  template <class Head, class Tail>
105  static ResultType DispatchLhs(BaseLhs& lhs, BaseRhs& rhs,
106  Executor exec, Typelist<Head, Tail>)
107  {
108  if (Head* p1 = dynamic_cast<Head*>(&lhs))
109  {
110  return DispatchRhs(*p1, rhs, exec, TypesRhs());
111  }
112  return DispatchLhs(lhs, rhs, exec, Tail());
113  }

116  static ResultType Go(BaseLhs& lhs, BaseRhs& rhs,
117  Executor exec)
118  { return DispatchLhs(lhs, rhs, exec, TypesLhs()); }

178  typename ResultType, typename CallbackType>

180  ::Go(BaseLhs& lhs, BaseRhs& rhs)
181  {
182  typename MapType::key_type k(typeid(lhs),typeid(rhs));
