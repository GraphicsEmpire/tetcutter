64 void TestBasic( A& a ) {
65  T x;
66  const T cx = T();
67 
68  // See Table 32 in ISO ++ Standard
69  typename A::pointer px = &x;
70  typename A::const_pointer pcx = &cx;
71 
72  typename A::reference rx = x;
73  ASSERT( &rx==&x, NULL );
74 
75  typename A::const_reference rcx = cx;
76  ASSERT( &rcx==&cx, NULL );
77 
78  typename A::value_type v = x;
79 
80  typename A::size_type size;
81  size = 0;
82  --size;
83  ASSERT( size>0, "not an unsigned integral type?" );
84 
85  typename A::difference_type difference;
86  difference = 0;
87  --difference;
88  ASSERT( difference<0, "not an signed integral type?" );
89 
90  // "rebind" tested by our caller
91 
92  ASSERT( a.address(rx)==px, NULL );
93 
94  ASSERT( a.address(rcx)==pcx, NULL );
95 
96  typename A::pointer array[100];
97  size_t sizeof_T = sizeof(T);
98  for( size_t k=0; k<100; ++k ) {
99  array[k] = k&1 ? a.allocate(k,array[0]) : a.allocate(k);
100  char* s = reinterpret_cast<char*>(reinterpret_cast<void*>(array[k]));
101  for( size_t j=0; j<k*sizeof_T; ++j )
102  s[j] = PseudoRandomValue(j,k);
103  }

105  // Test hint argument. This can't be compiled when hint is void*, It should be const void*
106  typename A::pointer a_ptr;
107  const void * const_hint = NULL;
108  a_ptr = a.allocate (1, const_hint);
109  a.deallocate(a_ptr, 1);

111  // Test "a.deallocate(p,n)
112  for( size_t k=0; k<100; ++k ) {
113  char* s = reinterpret_cast<char*>(reinterpret_cast<void*>(array[k]));
114  for( size_t j=0; j<k*sizeof_T; ++j )
115  ASSERT( s[j] == PseudoRandomValue(j,k), NULL );

160 // A is an allocator for some type
161 template<typename A>
162 struct Body: NoAssign {
163  static const size_t max_k = 100000;
164  A &a;
165  Body(A &a_) : a(a_) {}
166  void check_allocate( typename A::pointer array[], size_t i, size_t t ) const
167  {
168  ASSERT(array[i] == 0, NULL);
169  size_t size = i * (i&3);
170  array[i] = i&1 ? a.allocate(size, array[i>>3]) : a.allocate(size);
171  ASSERT(array[i] != 0, "allocator returned null");
172  char* s = reinterpret_cast<char*>(reinterpret_cast<void*>(array[i]));
173  for( size_t j=0; j<size*sizeof(typename A::value_type); ++j ) {
174  if(is_zero_filling<typename A::template rebind<void>::other>::value)
175  ASSERT( !s[j], NULL);
176  s[j] = PseudoRandomValue(i, t);
177  }
178  }
179 
180  void check_deallocate( typename A::pointer array[], size_t i, size_t t ) const
181  {
182  ASSERT(array[i] != 0, NULL);
183  size_t size = i * (i&3);
184  char* s = reinterpret_cast<char*>(reinterpret_cast<void*>(array[i]));
185  for( size_t j=0; j<size*sizeof(typename A::value_type); ++j )
186  ASSERT( s[j] == PseudoRandomValue(i, t), "Thread safety test failed" );
187  a.deallocate(array[i], size);
188  array[i] = 0;
189  }
190 
191  void operator()( size_t thread_id ) const {
192  typename A::pointer array[256];
193 
194  for( size_t k=0; k<256; ++k )
195  array[k] = 0;
196  for( size_t k=0; k<max_k; ++k ) {
197  size_t i = static_cast<unsigned char>(PseudoRandomValue(k,thread_id));
198  if(!array[i]) check_allocate(array, i, thread_id);
199  else check_deallocate(array, i, thread_id);
200  }
201  for( size_t k=0; k<256; ++k )
202  if(array[k])
203  check_deallocate(array, k, thread_id);
204  }
205 };
