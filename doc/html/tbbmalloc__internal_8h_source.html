140 template<typename Arg, typename Compare>
141 inline void AtomicUpdate(Arg &location, Arg newVal, const Compare &cmp)
142 {
143  MALLOC_STATIC_ASSERT(sizeof(Arg) == sizeof(intptr_t),
144  "Type of argument must match AtomicCompareExchange type.");
145  for (Arg old = location; cmp(old, newVal); ) {
146  Arg val = AtomicCompareExchange((intptr_t&)location, (intptr_t)newVal, old);
147  if (val == old)
148  break;
149  // TODO: do we need backoff after unsuccessful CAS?
150  old = val;
151  }
152 }

154 // TODO: make BitMaskBasic more general
155 // (currently, it fits BitMaskMin well, but not as suitable for BitMaskMax)
156 template<unsigned NUM>
157 class BitMaskBasic {
158  static const unsigned SZ = (NUM-1)/(CHAR_BIT*sizeof(uintptr_t))+1;
159  static const unsigned WORD_LEN = CHAR_BIT*sizeof(uintptr_t);
160  uintptr_t mask[SZ];
161 protected:
162  void set(size_t idx, bool val) {
163  MALLOC_ASSERT(idx<NUM, ASSERT_TEXT);
164 
165  size_t i = idx / WORD_LEN;
166  int pos = WORD_LEN - idx % WORD_LEN - 1;
167  if (val)
168  AtomicOr(&mask[i], 1ULL << pos);
169  else
170  AtomicAnd(&mask[i], ~(1ULL << pos));
171  }
172  int getMinTrue(unsigned startIdx) const {
173  unsigned idx = startIdx / WORD_LEN;
174  int pos;
175 
176  if (startIdx % WORD_LEN) {
177  // only interested in part of a word, clear bits before startIdx
178  pos = WORD_LEN - startIdx % WORD_LEN;
179  uintptr_t actualMask = mask[idx] & (((uintptr_t)1<<pos) - 1);
180  idx++;
181  if (-1 != (pos = BitScanRev(actualMask)))
182  return idx*WORD_LEN - pos - 1;
183  }
184 
185  while (idx<SZ)
186  if (-1 != (pos = BitScanRev(mask[idx++])))
187  return idx*WORD_LEN - pos - 1;
188  return -1;
189  }
190 public:
191  void reset() { for (unsigned i=0; i<SZ; i++) mask[i] = 0; }
192 };

563 // global state of blocks currently in processing
564 class BackendSync {
565  // Class instances should reside in zero-initialized memory!
566  // The number of blocks currently removed from a bin and not returned back
567  intptr_t blocksInProcessing; // to another
568  intptr_t binsModifications; // incremented on every bin modification
569 public:
570  void blockConsumed() { AtomicIncrement(blocksInProcessing); }
571  void binsModified() { AtomicIncrement(binsModifications); }
572  void blockReleased() {
573 #if __TBB_MALLOC_BACKEND_STAT
574  MALLOC_ITT_SYNC_RELEASING(&blocksInProcessing);
575 #endif
576  AtomicIncrement(binsModifications);
577  intptr_t prev = AtomicAdd(blocksInProcessing, -1);
578  MALLOC_ASSERT(prev > 0, ASSERT_TEXT);
579  suppress_unused_warning(prev);
580  }
581  intptr_t getNumOfMods() const { return FencedLoad(binsModifications); }
582  // return true if need re-do the blocks search
583  bool waitTillBlockReleased(intptr_t startModifiedCnt) {
584 #if __TBB_MALLOC_BACKEND_STAT
585  MALLOC_ITT_SYNC_PREPARE(&blocksInProcessing);
586 #endif
587  for (intptr_t myBlocksNum = FencedLoad(blocksInProcessing);
588  // no blocks in processing, stop waiting
589  myBlocksNum; ) {
590  SpinWaitWhileEq(blocksInProcessing, myBlocksNum);
591  WhiteboxTestingYield();
592  intptr_t newBlocksNum = FencedLoad(blocksInProcessing);
593  // stop waiting iff blocks were removed from processing,
594  // if blocks were added, there is no reason to stop waiting
595  if (newBlocksNum < myBlocksNum)
596  break;
597  myBlocksNum = newBlocksNum;
598  }
599 #if __TBB_MALLOC_BACKEND_STAT
600  MALLOC_ITT_SYNC_ACQUIRED(&blocksInProcessing);
601 #endif
602  // were bins modified since scanned?
603  return startModifiedCnt != getNumOfMods();
604  }
605 };

616  intptr_t active;
617 public:
618  bool wait() {
619  bool rescanBins = false;
620  // up to 3 threads can add more memory from OS simultaneously,
621  // rest of threads have to wait
622  for (;;) {
623  intptr_t prevCnt = FencedLoad(active);
624  if (prevCnt < 3) {
625  intptr_t n = AtomicCompareExchange(active, prevCnt+1, prevCnt);
626  if (n == prevCnt)
627  break;
628  } else {
629  SpinWaitWhileEq(active, prevCnt);
630  rescanBins = true;
631  break;
632  }
633  }
634  return rescanBins;
635  }
636  void signal() { AtomicAdd(active, -1); }
637 };
