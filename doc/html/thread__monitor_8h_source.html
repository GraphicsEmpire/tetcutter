49 // Deal with 64K aliasing. The formula for "offset" is a Fibonacci hash function,
50 // which has the desirable feature of spreading out the offsets fairly evenly
51 // without knowing the total number of offsets, and furthermore unlikely to
52 // accidentally cancel out other 64K aliasing schemes that Microsoft might implement later.
53 // See Knuth Vol 3. "Theorem S" for details on Fibonacci hashing.
54 // The second statement is really does need "volatile", otherwise the compiler might remove the _alloca.
55 #define AVOID_64K_ALIASING(idx) \
56  size_t offset = (idx+1) * 40503U % (1U<<16); \
57  void* volatile sink_for_alloca = _alloca(offset); \
58  __TBB_ASSERT_EX(sink_for_alloca, "_alloca failed");
59 #else
60 // Linux thread allocators avoid 64K aliasing.
61 #define AVOID_64K_ALIASING(idx) tbb::internal::suppress_unused_warning(idx)
62 #endif /* _WIN32||_WIN64 */

154 inline thread_monitor::handle_type thread_monitor::launch( thread_routine_type thread_routine, void* arg, size_t stack_size, const size_t* worker_index ) {
155  unsigned thread_id;
156  int number_of_processor_groups = ( worker_index ) ? tbb::internal::NumberOfProcessorGroups() : 0;
157  unsigned create_flags = ( number_of_processor_groups > 1 ) ? CREATE_SUSPENDED : 0;
158  HANDLE h = (HANDLE)_beginthreadex( NULL, unsigned(stack_size), thread_routine, arg, STACK_SIZE_PARAM_IS_A_RESERVATION | create_flags, &thread_id );
159  if( !h ) {
160  fprintf(stderr,"thread_monitor::launch: _beginthreadex failed\n");
161  exit(1);
162  }
163  if ( number_of_processor_groups > 1 ) {
164  tbb::internal::MoveThreadIntoProcessorGroup( h,
165  tbb::internal::FindProcessorGroupIndex( static_cast<int>(*worker_index) ) );
166  ResumeThread( h );
167  }
168  return h;
169 }

246 inline void thread_monitor::prepare_wait( cookie& c ) {
247  if( spurious ) {
248  spurious = false;
249  // consumes a spurious posted signal. don't wait on my_sema.
250  my_sema.P();

257 inline void thread_monitor::commit_wait( cookie& c ) {
258  bool do_it = ( c.my_epoch == my_cookie.my_epoch);
