200  void enable_segment( segment_index_t k, bool is_initial = false ) {
201  __TBB_ASSERT( k, "Zero segment must be embedded" );
202  enable_segment_failsafe watchdog( my_table, k );
203  cache_aligned_allocator<bucket> alloc;
204  size_type sz;
205  __TBB_ASSERT( !is_valid(my_table[k]), "Wrong concurrent assignment");
206  if( k >= first_block ) {
207  sz = segment_size( k );
208  segment_ptr_t ptr = alloc.allocate( sz );
209  init_buckets( ptr, sz, is_initial );
210  itt_hide_store_word( my_table[k], ptr );
211  sz <<= 1;// double it to get entire capacity of the container
212  } else { // the first block
213  __TBB_ASSERT( k == embedded_block, "Wrong segment index" );
214  sz = segment_size( first_block );
215  segment_ptr_t ptr = alloc.allocate( sz - embedded_buckets );
216  init_buckets( ptr, sz - embedded_buckets, is_initial );
217  ptr -= segment_base(embedded_block);
218  for(segment_index_t i = embedded_block; i < first_block; i++) // calc the offsets
219  itt_hide_store_word( my_table[i], ptr + segment_base(i) );
220  }
221  itt_store_word_with_release( my_mask, sz-1 );
222  watchdog.my_segment_ptr = 0;
223  }

255  bool check_rehashing_collision( const hashcode_t h, hashcode_t m_old, hashcode_t m ) const {
256  __TBB_ASSERT(m_old != m, NULL); // TODO?: m arg could be optimized out by passing h = h&m
257  if( (h & m_old) != (h & m) ) { // mask changed for this hashcode, rare event
258  // condition above proves that 'h' has some other bits set beside 'm_old'
259  // find next applicable mask after m_old //TODO: look at bsl instruction
260  for( ++m_old; !(h & m_old); m_old <<= 1 ) // at maximum few rounds depending on the first block size
261  ;
262  m_old = (m_old<<1) - 1; // get full mask from a bit
263  __TBB_ASSERT((m_old&(m_old+1))==0 && m_old <= m, NULL);
264  // check whether it is rehashing/ed
265  if( itt_load_word_with_acquire(get_bucket(h & m_old)->node_list) != rehash_req )
266  {
267 #if __TBB_STATISTICS
268  my_info_restarts++; // race collisions
269 #endif
270  return true;
271  }
272  }
273  return false;
274  }

1038 bool concurrent_hash_map<Key,T,HashCompare,A>::lookup( bool op_insert, const Key &key, const T *t, const_accessor *result, bool write, node* (*allocate_node)(node_allocator_type& , const Key&, const T*) ) {

1040  bool return_value;

1045  restart:
1046  {//lock scope
1047  __TBB_ASSERT((m&(m+1))==0, "data structure is invalid");
1048  return_value = false;
1049  // get bucket
1050  bucket_accessor b( this, h & m );
1051 
1052  // find a node
1053  n = search_bucket( key, b() );

1055  // [opt] insert a key
1056  if( !n ) {
1057  if( !tmp_n ) {
1058  tmp_n = allocate_node(my_allocator, key, t);
1059  }
1060  if( !b.is_writer() && !b.upgrade_to_writer() ) { // TODO: improved insertion
1061  // Rerun search_list, in case another thread inserted the item during the upgrade.
1062  n = search_bucket( key, b() );

1064  b.downgrade_to_reader();

1066  }
1067  }
1068  if( check_mask_race(h, m) )
1069  goto restart; // b.release() is done in ~b().

1071  grow_segment = insert_new_node( b(), n = tmp_n, m );
1072  tmp_n = 0;
1073  return_value = true;
1074  }
1075  } else { // find or count
1076  if( !n ) {
1077  if( check_mask_race( h, m ) )
1078  goto restart; // b.release() is done in ~b(). TODO: replace by continue
1079  return false;
1080  }
1081  return_value = true;
1082  }
1083  exists:
1084  if( !result ) goto check_growth;
1085  // TODO: the following seems as generic/regular operation
1086  // acquire the item

1088  for( tbb::internal::atomic_backoff backoff(true);; ) {
1089  if( result->try_acquire( n->mutex, write ) ) break;

1097  }
1098  }
1099  }
1100  }//lock scope
1101  result->my_node = n;
1102  result->my_hash = h;
1103 check_growth:
1104  // [opt] grow the container

1106 #if __TBB_STATISTICS
1107  my_info_resizes++; // concurrent ones
1108 #endif
1109  enable_segment( grow_segment );
1110  }
1111  if( tmp_n ) // if op_insert only
1112  delete_node( tmp_n );

1212 void concurrent_hash_map<Key,T,HashCompare,A>::rehash(size_type sz) {
1213  reserve( sz ); // TODO: add reduction of number of buckets as well
1214  hashcode_t mask = my_mask;
1215  hashcode_t b = (mask+1)>>1; // size or first index of the last segment
1216  __TBB_ASSERT((b&(b-1))==0, NULL); // zero or power of 2
1217  bucket *bp = get_bucket( b ); // only the last segment should be scanned for rehashing
1218  for(; b <= mask; b++, bp++ ) {
1219  node_base *n = bp->node_list;
1220  __TBB_ASSERT( is_valid(n) || n == internal::empty_rehashed || n == internal::rehash_req, "Broken internal structure" );
1221  __TBB_ASSERT( *reinterpret_cast<intptr_t*>(&bp->mutex) == 0, "concurrent or unexpectedly terminated operation during rehash() execution" );
1222  if( n == internal::rehash_req ) { // rehash bucket, conditional because rehashing of a previous bucket may affect this one
1223  hashcode_t h = b; bucket *b_old = bp;
1224  do {
1225  __TBB_ASSERT( h > 1, "The lowermost buckets can't be rehashed" );
1226  hashcode_t m = ( 1u<<__TBB_Log2( h ) ) - 1; // get parent mask from the topmost bit
1227  b_old = get_bucket( h &= m );
1228  } while( b_old->node_list == internal::rehash_req );
1229  // now h - is index of the root rehashed bucket b_old
1230  mark_rehashed_levels( h ); // mark all non-rehashed children recursively across all segments
1231  for( node_base **p = &b_old->node_list, *q = *p; is_valid(q); q = *p ) {
1232  hashcode_t c = my_hash_compare.hash( static_cast<node*>(q)->item.first );
1233  if( (c & mask) != h ) { // should be rehashed
1234  *p = q->next; // exclude from b_old
1235  bucket *b_new = get_bucket( c & mask );
1236  __TBB_ASSERT( b_new->node_list != internal::rehash_req, "hash() function changed for key in table or internal error" );
1237  add_to_bucket( b_new, q );
1238  } else p = &q->next; // iterate to next item
1239  }
1240  }
1241  }
1242 #if TBB_USE_PERFORMANCE_WARNINGS
1243  int current_size = int(my_size), buckets = int(mask)+1, empty_buckets = 0, overpopulated_buckets = 0; // usage statistics
1244  static bool reported = false;
1245 #endif
1246 #if TBB_USE_ASSERT || TBB_USE_PERFORMANCE_WARNINGS
1247  for( b = 0; b <= mask; b++ ) {// only last segment should be scanned for rehashing
1248  if( b & (b-2) ) ++bp; // not the beginning of a segment
