239 void WaitUntilConcurrencyPeaks ( int expected_peak ) {
240  if ( g_Flog )
241  return;
242  int n = 0;
243 retry:
244  while ( ++n < c_Timeout && (int)Harness::ConcurrencyTracker::PeakParallelism() < expected_peak )
245  __TBB_Yield();
246 #if USE_TASK_SCHEDULER_OBSERVER
247  ASSERT_WARNING( g_NumThreads == g_ActualMaxThreads, "Library did not provide sufficient threads");
248 #endif
249  ASSERT_WARNING(n < c_Timeout,g_Wakeup_Msg);
250  // Workaround in case a missed wakeup takes place
251  if ( n == c_Timeout ) {
252  tbb::task &r = *new( tbb::task::allocate_root() ) tbb::empty_task();
253  r.spawn(r);
254  n = 0;
255  goto retry;
256  }
257 }
