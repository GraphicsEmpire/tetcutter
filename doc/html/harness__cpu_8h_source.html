35 /* Returns 0 if not implemented on platform. */
36 static double GetCPUUserTime() {
37 #if _XBOX || __TBB_WIN8UI_SUPPORT
38  return 0;
39 #elif _WIN32
40  FILETIME my_times[4];
41  bool status = GetProcessTimes(GetCurrentProcess(), my_times, my_times+1, my_times+2, my_times+3)!=0;
42  ASSERT( status, NULL );
43  LARGE_INTEGER usrtime;
44  usrtime.LowPart = my_times[3].dwLowDateTime;
45  usrtime.HighPart = my_times[3].dwHighDateTime;
46  return double(usrtime.QuadPart)*1E-7;
47 #else
48  // Generic UNIX, including __APPLE__
49 
50  // On Linux, there is no good way to get CPU usage info for the current process:
51  // getrusage(RUSAGE_SELF, ...) that is used now only returns info for the calling thread;
52  // getrusage(RUSAGE_CHILDREN, ...) only counts for finished children threads;
53  // tms_utime and tms_cutime got with times(struct tms*) are equivalent to the above items;
54  // finally, /proc/self/task/<task_id>/stat doesn't exist on older kernels
55  // and it isn't quite convenient to read it for every task_id.
56 
57  struct rusage resources;
58  bool status = getrusage(RUSAGE_SELF, &resources)==0;
59  ASSERT( status, NULL );
60  return (double(resources.ru_utime.tv_sec)*1E6 + double(resources.ru_utime.tv_usec))*1E-6;
61 #endif
62 }

67 // The resolution of GetCPUUserTime is 10-15 ms or so; waittime should be a few times bigger.
68 const double WAITTIME = 0.1; // in seconds, i.e. 100 ms
69 const double THRESHOLD = WAITTIME/100;
70 
71 static void TestCPUUserTime( int nthreads, int nactive = 1 ) {
72  // The test will always pass on Linux; read the comments in GetCPUUserTime for details
73  // Also it will not detect spinning issues on systems with only one processing core.
74 
75  int nworkers = nthreads-nactive;
76  if( !nworkers ) return;
77  double lastusrtime = GetCPUUserTime();
78  if( !lastusrtime ) return;
79 
80  static double minimal_waittime = WAITTIME,
81  maximal_waittime = WAITTIME * 10;
82  double usrtime_delta;
83  double waittime_delta;
84  tbb::tick_count stamp = tbb::tick_count::now();
85  volatile intptr_t k = (intptr_t)&usrtime_delta;
86  // wait for GetCPUUserTime update
87  while( (usrtime_delta=GetCPUUserTime()-lastusrtime) < THRESHOLD ) {
88  for ( int i = 0; i < 1000; ++i ) ++k; // do fake work without which user time can stall
89  if ( (waittime_delta = (tbb::tick_count::now()-stamp).seconds()) > maximal_waittime ) {
90  REPORT( "Warning: %.2f sec elapsed but user mode time is still below its threshold (%g < %g)\n",
91  waittime_delta, usrtime_delta, THRESHOLD );
92  break;
93  }
94  }
95  lastusrtime += usrtime_delta;
96 
97  // Wait for workers to go sleep
98  stamp = tbb::tick_count::now();
99  while( ((waittime_delta=(tbb::tick_count::now()-stamp).seconds()) < minimal_waittime)
100  || ((usrtime_delta=GetCPUUserTime()-lastusrtime) < THRESHOLD) )
101  {
102  for ( int i = 0; i < 1000; ++i ) ++k; // do fake work without which user time can stall
103  if ( waittime_delta > maximal_waittime ) {
104  REPORT( "Warning: %.2f sec elapsed but GetCPUUserTime reported only %g sec\n", waittime_delta, usrtime_delta );
105  break;
106  }
107  }
108 
109  // Test that all workers sleep when no work.
110  while( nactive>1 && usrtime_delta-nactive*waittime_delta<0 ) {
111  // probably the number of active threads was mispredicted
112  --nactive; ++nworkers;
113  }
114  double avg_worker_usrtime = (usrtime_delta-nactive*waittime_delta)/nworkers;
115 
116  if( avg_worker_usrtime > waittime_delta/2 )
117  REPORT( "ERROR: %d worker threads are spinning; waittime: %g; usrtime: %g; avg worker usrtime: %g\n",
118  nworkers, waittime_delta, usrtime_delta, avg_worker_usrtime);
119  else
120  REMARK("%d worker threads; waittime: %g; usrtime: %g; avg worker usrtime: %g\n",
121  nworkers, waittime_delta, usrtime_delta, avg_worker_usrtime);
122 }
