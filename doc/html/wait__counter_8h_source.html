37  my_count=1;
38  n_transients=0;
39  }

43  void wait() {
44  int k = --my_count;
45  __TBB_ASSERT( k>=0, "counter underflow" );
46  if( k>0 ) {
47  thread_monitor::cookie c;
48  my_monitor.prepare_wait(c);
49  if( my_count )
50  my_monitor.commit_wait(c);
51  else
52  my_monitor.cancel_wait();
53  }
54  while( n_transients>0 )
55  __TBB_Yield();
56  }
57  void operator++() {
58  ++my_count;
59  }
60  void operator--() {
61  ++n_transients;
62  int k = --my_count;
63  __TBB_ASSERT( k>=0, "counter underflow" );
64  if( k==0 )
65  my_monitor.notify();
66  --n_transients;
67  }
68 };
