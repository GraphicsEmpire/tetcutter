74  template<class F> struct thread_closure_0: thread_closure_base {
75  F function;

77  static __TBB_NATIVE_THREAD_ROUTINE start_routine( void* c ) {
78  thread_closure_0 *self = static_cast<thread_closure_0*>(c);
79  self->function();
80  delete self;
81  return 0;
82  }
83  thread_closure_0( const F& f ) : function(f) {}
84  };
86  template<class F, class X> struct thread_closure_1: thread_closure_base {
87  F function;
88  X arg1;
90  static __TBB_NATIVE_THREAD_ROUTINE start_routine( void* c ) {
91  thread_closure_1 *self = static_cast<thread_closure_1*>(c);
92  self->function(self->arg1);
93  delete self;
94  return 0;
95  }
96  thread_closure_1( const F& f, const X& x ) : function(f), arg1(x) {}
97  };
98  template<class F, class X, class Y> struct thread_closure_2: thread_closure_base {
99  F function;
100  X arg1;
101  Y arg2;
103  static __TBB_NATIVE_THREAD_ROUTINE start_routine( void* c ) {
104  thread_closure_2 *self = static_cast<thread_closure_2*>(c);
105  self->function(self->arg1, self->arg2);
106  delete self;
107  return 0;
108  }
109  thread_closure_2( const F& f, const X& x, const Y& y ) : function(f), arg1(x), arg2(y) {}
110  };
