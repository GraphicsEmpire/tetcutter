68  bool try_acquire() {
69  intptr_t snapshot = my_job;
70  if( snapshot==-1 ) {
71  return false;
72  } else {
73  __TBB_ASSERT( (snapshot&1)==0, "already marked that way" );
74  intptr_t old = my_job.compare_and_swap( snapshot|1, snapshot );
75  __TBB_ASSERT( old==snapshot || old==-1, "unexpected interference" );
76  return old==snapshot;
77  }
78  }

81  void release() {
82  intptr_t snapshot = my_job;
83  __TBB_ASSERT( snapshot&1, NULL );
84  // Atomic store suffices here.
85  my_job = snapshot&~1;
86  }

107  bool try_plug( rml::job*&j ) {
108  for(;;) {
109  intptr_t snapshot = my_job;
110  if( snapshot&1 ) {
111  j = NULL;
112  return false;
113  }
114  // Not busy
115  if( my_job.compare_and_swap( -1, snapshot )==snapshot ) {
116  j = reinterpret_cast<rml::job*>(snapshot);
117  return true;
118  }
119  // Need to retry, because current thread may be non-owner that read a 0, and owner might have
120  // caused transition 0->1->ptr after we took our snapshot.
121  }
122  }

125  rml::job& wait_for_job() const {
126  intptr_t snapshot;
127  for(;;) {
128  snapshot = my_job;
129  if( snapshot&~1 ) break;
130  __TBB_Yield();
131  }
132  __TBB_ASSERT( snapshot!=-1, "wait on plugged job_automaton" );
133  return *reinterpret_cast<rml::job*>(snapshot&~1);
134  }
