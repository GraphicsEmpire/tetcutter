74  public:
75  template<class test_class>
76  void register_test_case(std::string const& name, test_class * ){
77  test_cases.push_back(tc_record_pair(name,& run_test<test_class>));
78  }
79  std::string operator()(bool silent=false){
80  std::stringstream str;
81  size_t failed=0;
82  for (size_t i=0;i<test_cases.size();++i){
83  try{
84  (test_cases[i].second)();
85  }catch(std::exception& e){
86  failed++;
87  str<<"test case \""<<test_cases[i].first<<"\" failed with exception. what():\""<<e.what()<<"\""<<std::endl;
88  }
89  }
90  if (!silent) {
91  str<<test_cases.size()<<" test cases are run; "<<failed<<" failed"<<std::endl;
92  }
93  return str.str();
94  }
