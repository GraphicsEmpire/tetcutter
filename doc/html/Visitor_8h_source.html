252  template<typename R,template <typename, class> class CatchAll>
253  class BaseVisitable<R, CatchAll, false>
254  {
255  public:

257  virtual ~BaseVisitable() {}

260  protected: // give access only to the hierarchy
261  template <class T>
262  static ReturnType AcceptImpl(T& visited, BaseVisitor& guest)
263  {
264  // Apply the Acyclic Visitor
265  if (Visitor<T,R>* p = dynamic_cast<Visitor<T,R>*>(&guest))
