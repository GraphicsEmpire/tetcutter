275 template<bool is_master>
276 inline void arena::on_thread_leaving ( ) {
277  //
278  // Implementation of arena destruction synchronization logic contained various
279  // bugs/flaws at the different stages of its evolution, so below is a detailed
280  // description of the issues taken into consideration in the framework of the
281  // current design.
282  //
283  // In case of using fire-and-forget tasks (scheduled via task::enqueue())
284  // master thread is allowed to leave its arena before all its work is executed,
285  // and market may temporarily revoke all workers from this arena. Since revoked
286  // workers never attempt to reset arena state to EMPTY and cancel its request
287  // to RML for threads, the arena object is destroyed only when both the last
288  // thread is leaving it and arena's state is EMPTY (that is its master thread
289  // left and it does not contain any work).
290  //
291  // A worker that checks for work presence and transitions arena to the EMPTY
292  // state (in snapshot taking procedure arena::is_out_of_work()) updates
293  // arena::my_pool_state first and only then arena::my_num_workers_requested.
294  // So the check for work absence must be done against the latter field.
295  //
296  // In a time window between decrementing the active threads count and checking
297  // if there is an outstanding request for workers. New worker thread may arrive,
298  // finish remaining work, set arena state to empty, and leave decrementing its
299  // refcount and destroying. Then the current thread will destroy the arena
300  // the second time. To preclude it a local copy of the outstanding request
301  // value can be stored before decrementing active threads count.
302  //
303  // But this technique may cause two other problem. When the stored request is
304  // zero, it is possible that arena still has threads and they can generate new
305  // tasks and thus re-establish non-zero requests. Then all the threads can be
306  // revoked (as described above) leaving this thread the last one, and causing
307  // it to destroy non-empty arena.
308  //
309  // The other problem takes place when the stored request is non-zero. Another
310  // thread may complete the work, set arena state to empty, and leave without
311  // arena destruction before this thread decrements the refcount. This thread
312  // cannot destroy the arena either. Thus the arena may be "orphaned".
313  //
314  // In both cases we cannot dereference arena pointer after the refcount is
315  // decremented, as our arena may already be destroyed.
316  //
317  // If this is the master thread, market can be concurrently destroyed.
318  // In case of workers market's liveness is ensured by the RML connection
319  // rundown protocol, according to which the client (i.e. the market) lives
320  // until RML server notifies it about connection termination, and this
321  // notification is fired only after all workers return into RML.
322  //
323  // Thus if we decremented refcount to zero we ask the market to check arena
324  // state (including the fact if it is alive) under the lock.
325  //
326  uintptr_t aba_epoch = my_aba_epoch;
327  market* m = my_market;
328  __TBB_ASSERT(my_references > int(!is_master), "broken arena reference counter");
329  if ( (my_references -= is_master? 1:2 ) == 0 ) // worker's counter starts from bit 1
330  market::try_destroy_arena( m, this, aba_epoch, is_master );
331 }

333 template<bool Spawned> void arena::advertise_new_work() {
334  if( !Spawned ) { // i.e. the work was enqueued
335  if( my_max_num_workers==0 ) {
336  my_max_num_workers = 1;
337  __TBB_ASSERT(!my_mandatory_concurrency, "");
338  my_mandatory_concurrency = true;
339  __TBB_ASSERT(!num_workers_active(), "");
340  my_pool_state = SNAPSHOT_FULL;
341  my_market->adjust_demand( *this, 1 );
342  return;
343  }
344  // Local memory fence is required to avoid missed wakeups; see the comment below.
345  // Starvation resistant tasks require mandatory concurrency, so missed wakeups are unacceptable.
346  atomic_fence();
347  }
348  // Double-check idiom that, in case of spawning, is deliberately sloppy about memory fences.
349  // Technically, to avoid missed wakeups, there should be a full memory fence between the point we
350  // released the task pool (i.e. spawned task) and read the arena's state. However, adding such a
351  // fence might hurt overall performance more than it helps, because the fence would be executed
352  // on every task pool release, even when stealing does not occur. Since TBB allows parallelism,
353  // but never promises parallelism, the missed wakeup is not a correctness problem.
354  pool_state_t snapshot = my_pool_state;
355  if( is_busy_or_empty(snapshot) ) {
356  // Attempt to mark as full. The compare_and_swap below is a little unusual because the
357  // result is compared to a value that can be different than the comparand argument.
358  if( my_pool_state.compare_and_swap( SNAPSHOT_FULL, snapshot )==SNAPSHOT_EMPTY ) {
359  if( snapshot!=SNAPSHOT_EMPTY ) {
360  // This thread read "busy" into snapshot, and then another thread transitioned
361  // my_pool_state to "empty" in the meantime, which caused the compare_and_swap above
362  // to fail. Attempt to transition my_pool_state from "empty" to "full".
363  if( my_pool_state.compare_and_swap( SNAPSHOT_FULL, SNAPSHOT_EMPTY )!=SNAPSHOT_EMPTY ) {
364  // Some other thread transitioned my_pool_state from "empty", and hence became
365  // responsible for waking up workers.
366  return;
367  }
368  }
369  // This thread transitioned pool from empty to full state, and thus is responsible for
370  // telling RML that there is work to do.
371  if( Spawned ) {
372  if( my_mandatory_concurrency ) {
373  __TBB_ASSERT(my_max_num_workers==1, "");
374  __TBB_ASSERT(!governor::local_scheduler()->is_worker(), "");
375  // There was deliberate oversubscription on 1 core for sake of starvation-resistant tasks.
376  // Now a single active thread (must be the master) supposedly starts a new parallel region
377  // with relaxed sequential semantics, and oversubscription should be avoided.
378  // Demand for workers has been decreased to 0 during SNAPSHOT_EMPTY, so just keep it.
379  my_max_num_workers = 0;
380  my_mandatory_concurrency = false;
381  return;
382  }
383  }
384  my_market->adjust_demand( *this, my_max_num_workers );
385  }
386  }
387 }
