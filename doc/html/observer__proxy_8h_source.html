140  if( p->my_observer ) {
141  // Can decrement refcount quickly, as it cannot drop to zero while under the lock.
142  int r = --p->my_ref_count;
143  __TBB_ASSERT_EX( r, NULL );
144  p = NULL;
145  } else {
146  // Use slow form of refcount decrementing, after the lock is released.

156 inline void observer_list::notify_exit_observers( observer_proxy*& last, bool worker ) {
157  if ( !last )
158  return;
159  __TBB_ASSERT(is_alive((uintptr_t)last), NULL);
160  do_notify_exit_observers( last, worker );
161  __TBB_ASSERT(last, NULL);
162  poison_value(last);
