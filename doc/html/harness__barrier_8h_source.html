83  // onOpenBarrierCallback is called by last thread arrived on a barrier
84  template<typename WaitEq, typename Callback>
85  bool custom_wait(const WaitEq &onWaitCallback, const Callback &onOpenBarrierCallback)
86  { // return true if last thread
87  unsigned myEpoch = epoch;
88  int threadsLeft = numThreads - numThreadsFinished.fetch_and_increment() - 1;
89  ASSERT(threadsLeft>=0, "Broken barrier");
90  if (threadsLeft > 0) {
91  /* not the last threading reaching barrier, wait until epoch changes & return 0 */
92  onWaitCallback(epoch, myEpoch);
93  return false;
94  }
95  onOpenBarrierCallback();
96  /* No more threads left to enter, so I'm the last one reaching this epoch;
97  reset the barrier, increment epoch, and return non-zero */
98  threadsLeft = numThreadsFinished -= numThreads;
99  ASSERT( threadsLeft == 0, "Broken barrier");
100  /* wakes up threads waiting to exit this epoch */
101  myEpoch -= epoch++;
102  ASSERT( myEpoch == 0, "Broken barrier");
103  return true;
104  }
105  bool timed_wait(double n_seconds, const char *msg="Time is out while waiting on a barrier") {
106  bool is_last = custom_wait(TimedWaitWhileEq(n_seconds), DummyCallback());
107  ASSERT( n_seconds >= 0, msg); // TODO: refactor to avoid passing msg here and rising assertion
108  return is_last;
109  }
