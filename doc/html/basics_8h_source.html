185  bool operator()(signal_t& out) {

202  clock_node(g, clock_body(ms, reps), false)
203  {}

205  my_graph(src.my_graph), ms(src.init_ms), init_ms(src.init_ms),

310 class and_gate : public gate<N> {
311  using gate<N>::my_graph;
312  typedef typename gate<N>::ports_type ports_type;
313  typedef typename gate<N>::input_port_t::output_type from_input;
314  class and_body {
315  signal_t *ports;
316  signal_t state;
317  bool touched;
318  public:
319  and_body() : state(undefined), touched(false) {
320  ports = new signal_t[N];
321  for (int i=0; i<N; ++i) ports[i] = undefined;
322  }
323  void operator()(const from_input& v, ports_type& p) {
324  ports[v.tag()] = cast_to<signal_t>(v);
325  signal_t new_state=high;
326  size_t i=0;
327  while (i<N) {
328  if (ports[i] == low) { new_state = low; break; }
329  else if (ports[i] == undefined && new_state != low) { new_state = undefined; }
330  ++i;
331  }
332  if (!touched || state != new_state) {
333  state = new_state;
334  tbb::flow::get<0>(p).try_put(state);
335  touched = true;
336  }
337  }
338  };
339  public:
340  and_gate(graph& g) : gate<N>(g, and_body()) {}
