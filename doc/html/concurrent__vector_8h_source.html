197  concurrent_vector_base_v3() {
198  //Here the semantic is intentionally relaxed.
199  //The reason this is next:
200  //Object that is in middle of construction (i.e. its constructor is not yet finished)
201  //cannot be used concurrently until the construction is finished.
202  //Thus to flag other threads that construction is finished, some synchronization with
203  //acquire-release semantic should be done by the (external) code that uses the vector.
204  //So, no need to do the synchronization inside the vector.
205 
206  my_early_size.store<relaxed>(0);
207  my_first_block.store<relaxed>(0); // here is not default_initial_segments
208  my_segment.store<relaxed>(my_storage);
209  }

1190 void concurrent_vector<T, A>::internal_free_segments(segment_t table[], segment_index_t k, segment_index_t first_block) {
1191  // Free the arrays
1192  while( k > first_block ) {
1193  --k;
1194  segment_value_t segment_value = table[k].load<relaxed>();
1195  table[k].store<relaxed>(segment_not_used());
1196  if( segment_value == segment_allocated() ) // check for correct segment pointer
1197  this->my_allocator.deallocate( (segment_value.pointer<T>()), segment_size(k) );
1198  }
1199  segment_value_t segment_value = table[0].load<relaxed>();
1200  if( segment_value == segment_allocated() ) {
1201  __TBB_ASSERT( first_block > 0, NULL );
1202  while(k > 0) table[--k].store<relaxed>(segment_not_used());
1203  this->my_allocator.deallocate( (segment_value.pointer<T>()), segment_size(first_block) );
1204  }
1205 }

1243 template<typename T, class A> template<class I>
1244 void concurrent_vector<T, A>::internal_assign_iterators(I first, I last) {
1245  __TBB_ASSERT(my_early_size == 0, NULL);
1246  size_type n = std::distance(first, last);
1247  if( !n ) return;
1248  internal_reserve(n, sizeof(T), max_size());
1249  my_early_size = n;
1250  segment_index_t k = 0;
1251  //TODO: unify segment iteration code with concurrent_base_v3::helper
1252  size_type sz = segment_size( my_first_block );
1253  while( sz < n ) {
1254  internal_loop_guide loop(sz, my_segment[k].template load<relaxed>().template pointer<void>());
1255  loop.iterate(first);
1256  n -= sz;
1257  if( !k ) k = my_first_block;
1258  else { ++k; sz <<= 1; }
1259  }
1260  internal_loop_guide loop(n, my_segment[k].template load<relaxed>().template pointer<void>());
1261  loop.iterate(first);
1262 }
