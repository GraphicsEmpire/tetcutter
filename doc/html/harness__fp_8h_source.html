70 inline void SetRoundingMode ( int mode ) {
71  tbb::internal::cpu_ctl_env ctl;
72  ctl.get_env();
73  ctl.mxcsr = (ctl.mxcsr & ~SSE_RND_MODE_MASK) | (mode & FE_RND_MODE_MASK) << 3;
74  ctl.x87cw = short((ctl.x87cw & ~FE_RND_MODE_MASK) | (mode & FE_RND_MODE_MASK));
75  ctl.set_env();
