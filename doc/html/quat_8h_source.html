120 void Quaternion<T>::fromEuler(T roll, T pitch, T yaw)
121 {
122  roll *= 0.5f;
123  pitch *= 0.5f;
124  yaw *= 0.5f;
125 
126  T cr = (T)cos(roll);
127  T cp = (T)cos(pitch);
128  T cy = (T)cos(yaw);
129 
130  T sr = (T)sin(roll);
131  T sp = (T)sin(pitch);
132  T sy = (T)sin(yaw);
133 
134  T cpcy = cp * cy;
135  T spsy = sp * sy;
136  T spcy = sp * cy;
137  T cpsy = cp * sy;
138 
139  w = cr * cpcy + sr * spsy;
140  x = sr * cpcy - cr * spsy;
141  y = cr * spcy + sr * cpsy;
142  z = cr * cpsy - sr * spcy;
143 }

145 template <typename T>
146 void Quaternion<T>::toEuler(T& roll, T& pitch, T& yaw ) const
147 {
148  T sint = (2.0f * w * y) - (2.0f * x * z);
149  T cost_temp = 1.0f - (sint * sint);
150  T cost = 0;
151 
152  if ( (T)fabs(cost_temp) > 0.001f )
153  {
154  cost = sqrt( cost_temp );
155  }
156 
157  T sinv, cosv, sinf, cosf;
158  if ( (T)fabs(cost) > 0.001f )
159  {
160  sinv = ((2.0f * y * z) + (2.0f * w * x)) / cost;
161  cosv = (1.0f - (2.0f * x * x) - (2.0f * y * y)) / cost;
162  sinf = ((2.0f * x * y) + (2.0f * w * z)) / cost;
163  cosf = (1.0f - (2.0f * y * y) - (2.0f * z * z)) / cost;
164  }
165  else
166  {
167  sinv = (2.0f * w * x) - (2.0f * y * z);
168  cosv = 1.0f - (2.0f * x * x) - (2.0f * z * z);
169  sinf = 0;
170  cosf = 1.0f;
171  }
172 
173  // compute output rotations
174  roll = atan2( sinv, cosv );
175  pitch = atan2( sint, cost );
176  yaw = atan2( sinf, cosf );
177 }
