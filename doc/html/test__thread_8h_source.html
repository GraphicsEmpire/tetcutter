116 void CheckRelations( const THREAD::id ids[], int n, bool duplicates_allowed ) {
117  for( int i=0; i<n; ++i ) {
118  const THREAD::id x = ids[i];
119  for( int j=0; j<n; ++j ) {
120  const THREAD::id y = ids[j];
121  ASSERT( (x==y)==!(x!=y), NULL );
122  ASSERT( (x<y)==!(x>=y), NULL );
123  ASSERT( (x>y)==!(x<=y), NULL );
124  ASSERT( (x<y)+(x==y)+(x>y)==1, NULL );
125  ASSERT( x!=y || i==j || duplicates_allowed, NULL );
126  for( int k=0; k<n; ++k ) {
127  const THREAD::id z = ids[j];
128  ASSERT( !(x<y && y<z) || x<z, "< is not transitive" );
129  }
130  }
131  }

144  int original_count = BaseCount;
145  // d loops over number of copies before throw occurs
146  for( int d=1; d<=3; ++d ) {
147  // Try all combinations of throw/nothrow for f, x, and y's copy constructor.
148  for( int i=0; i<8; ++i ) {
149  {
150  const AnotherThreadFunc f = AnotherThreadFunc();
151  if( i&1 ) f.copy_throws = d;
152  const Data<1> x(0);
153  if( i&2 ) x.copy_throws = d;
154  const Data<2> y(0);
155  if( i&4 ) y.copy_throws = d;
156  bool exception_caught = false;
157  for( int j=0; j<3; ++j ) {
158  try {
159  switch(j) {
160  case 0: {THREAD t(f); t.join();} break;
161  case 1: {THREAD t(f,x); t.join();} break;
162  case 2: {THREAD t(f,x,y); t.join();} break;
163  }
164  } catch(...) {
165  exception_caught = true;
166  }
167  ASSERT( !exception_caught||(i&((1<<(j+1))-1))!=0, NULL );
168  }
169  }

171  }
