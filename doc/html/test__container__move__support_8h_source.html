41  DefaultInitialized =0xDEFAUL,

44  MoveInitialized =0xAAAAA,

47  MovedFrom =0xFFFFF,
48  Destroyed =0xDEADF00
49  };

52  template<bool allow_zero_initialized_state = false>
53  struct StateTrackable: StateTrackableBase{
54  static const bool is_zero_initialized_state_allowed = allow_zero_initialized_state;
55  State state;
56 
57  bool is_valid() const{
58  return state == DefaultInitialized || state == DirectInitialized || state == CopyInitialized
59  || state == MoveInitialized || state == Assigned || state == MoveAssigned || state == MovedFrom
60  || (allow_zero_initialized_state && state == ZeroInitialized)
61  ;
62  }
63 
64  StateTrackable (intptr_t) __TBB_NOTHROW : state (DirectInitialized){}
65  StateTrackable () __TBB_NOTHROW : state (DefaultInitialized){}
66  StateTrackable (const StateTrackable & src) __TBB_NOTHROW{
67  ASSERT( src.is_valid(), "bad source for copy" );
68  state = CopyInitialized;
69  }
70  #if __TBB_CPP11_RVALUE_REF_PRESENT
71  StateTrackable (StateTrackable && src) __TBB_NOTHROW{
72  ASSERT( src.is_valid(), "bad source for move?" );
73  state = MoveInitialized;
74  src.state = MovedFrom;
75  }
76  StateTrackable & operator=(StateTrackable && src) __TBB_NOTHROW{
77  ASSERT( src.is_valid(), "bad source for assignment" );
78  ASSERT( is_valid(), "assigning to invalid instance?" );
79 
80  src.state = MovedFrom;
81  state = MoveAssigned;
82  return *this;
83  }
84  #endif
85  StateTrackable & operator=(const StateTrackable & src) __TBB_NOTHROW{
86  ASSERT( src.is_valid(), "bad source for assignment?" );
87  ASSERT( is_valid(), "assigning to invalid instance?" );
88 
89  state = Assigned;
90  return *this;
91  }
92  ~StateTrackable () __TBB_NOTHROW{
93  ASSERT( is_valid(), "Calling destructor on invalid instance? (twice destructor call?)" );
94  state = Destroyed;
95  }
96  };
