59  void __TBB_EXPORTED_FUNC assertion_failure( const char* filename, int line, const char* expression, const char* comment ) {
60  if( assertion_handler_type a = assertion_handler ) {
61  (*a)(filename,line,expression,comment);
62  } else {
63  static bool already_failed;
64  if( !already_failed ) {
65  already_failed = true;
66  fprintf( stderr, "Assertion %s failed on line %d of file %s\n",
67  expression, line, filename );
68  if( comment )
69  fprintf( stderr, "Detailed description: %s\n", comment );
70 #if _MSC_VER && _DEBUG
71  if(1 == _CrtDbgReport(_CRT_ASSERT, filename, line, "tbb_debug.dll", "%s\r\n%s", expression, comment?comment:""))
