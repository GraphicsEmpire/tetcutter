59 namespace tbb {
60 namespace internal {
61 class cpu_ctl_env {
62 private:
63  int mxcsr;
64  short x87cw;
65  static const int MXCSR_CONTROL_MASK = ~0x3f; /* all except last six status bits */
66 public:
67  bool operator!=( const cpu_ctl_env& ctl ) const { return mxcsr != ctl.mxcsr || x87cw != ctl.x87cw; }
68  void get_env() {
69  #if __TBB_ICC_12_0_INL_ASM_FSTCW_BROKEN
70  cpu_ctl_env loc_ctl;
71  __asm__ __volatile__ (
72  "stmxcsr %0\n\t"
73  "fstcw %1"
74  : "=m"(loc_ctl.mxcsr), "=m"(loc_ctl.x87cw)
75  );
76  *this = loc_ctl;
77  #else
78  __asm__ __volatile__ (
79  "stmxcsr %0\n\t"
80  "fstcw %1"
81  : "=m"(mxcsr), "=m"(x87cw)
82  );
83  #endif
84  mxcsr &= MXCSR_CONTROL_MASK;
85  }
86  void set_env() const {
87  __asm__ __volatile__ (
88  "ldmxcsr %0\n\t"
89  "fldcw %1"
90  : : "m"(mxcsr), "m"(x87cw)
91  );
92  }
93 };
94 } // namespace internal
95 } // namespace tbb
