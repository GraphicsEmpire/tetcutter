94  void acquire( spin_rw_mutex& m, bool write = true ) {
95  __TBB_ASSERT( !mutex, "holding mutex already" );
96  mutex = &m;
97  is_writer = write;
98  if( write ) internal_acquire_writer(mutex);
99  else internal_acquire_reader(mutex);
100  }

104  bool upgrade_to_writer() {
105  __TBB_ASSERT( mutex, "lock is not acquired" );
106  __TBB_ASSERT( !is_writer, "not a reader" );
107  is_writer = true;
108  return internal_upgrade(mutex);
109  }

112  void release() {
113  __TBB_ASSERT( mutex, "lock is not acquired" );
114  spin_rw_mutex *m = mutex;
115  mutex = NULL;
116  if( is_writer ) {
117 #if TBB_USE_THREADING_TOOLS||TBB_USE_ASSERT
118  internal_release_writer(m);
119 #else
120  m->state = 0;
121 #endif /* TBB_USE_THREADING_TOOLS||TBB_USE_ASSERT */
122  } else {
123  internal_release_reader(m);
124  }
125  };
126 
128  bool downgrade_to_reader() {
129  __TBB_ASSERT( mutex, "lock is not acquired" );
130  __TBB_ASSERT( is_writer, "not a writer" );
131 #if TBB_USE_THREADING_TOOLS||TBB_USE_ASSERT
132  internal_downgrade(mutex);
133 #else
134  mutex->state = 4; // Bit 2 - reader, 00..00100
135 #endif
136  is_writer = false;
137  return true;
138  }
