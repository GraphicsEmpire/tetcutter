128 template<typename M>
129 void Counter<M>::flog_once_lock_guard(size_t mode)
131 {
132  if( mode&1 ) {
133  // Try acquire and release with implicit lock_guard
134  // precondition: if mutex_type is not a recursive mutex, the calling thread does not own the mutex m.
135  // if the precondition is not met, either dead-lock incorrect 'value' would result in.
136  lock_guard<M> lg(mutex);
137  value = value+1;
138  } else {
139  // Try acquire and release with adopt lock_quard
140  // precodition: the calling thread owns the mutex m.
141  // if the precondition is not met, incorrect 'value' would result in because the thread unlocks
142  // mutex that it does not own.
143  mutex.lock();
144  lock_guard<M> lg( mutex, adopt_lock );
145  value = value+1;
146  }
147 }

149 template<typename M>
150 void Counter<M>::flog_once_unique_lock(size_t mode)
152 {
153  switch( mode&7 ) {
154  case 0:
155  {// implicitly acquire and release mutex with unique_lock
156  unique_lock<M> ul( mutex );
157  value = value+1;
158  ASSERT( ul==true, NULL );
159  }
160  break;
161  case 1:
162  {// unique_lock with defer_lock
163  unique_lock<M> ul( mutex, defer_lock );
164  ASSERT( ul.owns_lock()==false, NULL );
165  ul.lock();
166  value = value+1;
167  ASSERT( ul.owns_lock()==true, NULL );
168  }
169  break;
170  case 2:
171  {// unique_lock::try_lock() with try_to_lock
172  unique_lock<M> ul( mutex, try_to_lock );
173  if( !ul )
174  while( !ul.try_lock() )
175  __TBB_Yield();
176  value = value+1;
177  }
178  break;
179  case 3:
180  {// unique_lock::try_lock_for() with try_to_lock
181  unique_lock<M> ul( mutex, defer_lock );
182  tbb::tick_count::interval_t i(1.0);
183  while( !ul.try_lock_for( i ) )
184  ;
185  value = value+1;
186  ASSERT( ul.owns_lock()==true, NULL );
187  }
188  break;
189  case 4:
190  {
191  unique_lock<M> ul_o4;
192  {// unique_lock with adopt_lock
193  mutex.lock();
194  unique_lock<M> ul( mutex, adopt_lock );
195  value = value+1;
196  ASSERT( ul.owns_lock()==true, NULL );
197  ASSERT( ul.mutex()==&mutex, NULL );
198  ASSERT( ul_o4.owns_lock()==false, NULL );
199  ASSERT( ul_o4.mutex()==NULL, NULL );
200  swap( ul, ul_o4 );
201  ASSERT( ul.owns_lock()==false, NULL );
202  ASSERT( ul.mutex()==NULL, NULL );
203  ASSERT( ul_o4.owns_lock()==true, NULL );
204  ASSERT( ul_o4.mutex()==&mutex, NULL );
205  ul_o4.unlock();
206  }
207  ASSERT( ul_o4.owns_lock()==false, NULL );
208  }
209  break;
210  case 5:
211  {
212  unique_lock<M> ul_o5;
213  {// unique_lock with adopt_lock
214  mutex.lock();
215  unique_lock<M> ul( mutex, adopt_lock );
216  value = value+1;
217  ASSERT( ul.owns_lock()==true, NULL );
218  ASSERT( ul.mutex()==&mutex, NULL );
219  ASSERT( ul_o5.owns_lock()==false, NULL );
220  ASSERT( ul_o5.mutex()==NULL, NULL );
221  ul_o5.swap( ul );
222  ASSERT( ul.owns_lock()==false, NULL );
223  ASSERT( ul.mutex()==NULL, NULL );
224  ASSERT( ul_o5.owns_lock()==true, NULL );
225  ASSERT( ul_o5.mutex()==&mutex, NULL );
226  ul_o5.unlock();
227  }
228  ASSERT( ul_o5.owns_lock()==false, NULL );
229  }
230  break;
231  default:
232  {// unique_lock with adopt_lock, and release()
233  mutex.lock();
234  unique_lock<M> ul( mutex, adopt_lock );
235  ASSERT( ul==true, NULL );
236  value = value+1;
237  M* old_m = ul.release();
238  old_m->unlock();
239  ASSERT( ul.owns_lock()==false, NULL );
240  }
241  break;
242  }
243 }

509 struct WorkForCondVarWaitAndNotifyOne: NoAssign {
510  int nthread;
511  condition_variable& test_cv;
512  M& my_mtx;
513  WorkForCondVarWaitAndNotifyOne( int n_, condition_variable& cv_, M& mtx_ ) : nthread(n_), test_cv(cv_), my_mtx(mtx_) {}
514  void operator()( int tid ) const {
515  if( tid&1 ) {
516  // exercise signal part
517  while( ticket_for_wakeup<max_ticket ) {
518  int my_ticket = ++ticket_for_wakeup; // atomically grab the next ticket
519  if( my_ticket>max_ticket )
520  break;
521 
522  for( ;; ) {
523  unique_lock<M> ul( my_mtx, defer_lock );
524  ul.lock();
525  if( n_waiters>0 && my_ticket<=ticket_for_sleep && my_ticket==(wokeup_ticket+1) ) {
526  signaled_ticket = my_ticket;
527  test_cv.notify_one();
528  ++n_signaled;
529  ul.unlock();
530  break;
531  }
532  ul.unlock();
533  __TBB_Yield();
534  }
535 
536  // give waiters time to go to sleep.
537  for( int m=0; m<short_delay; ++m )
538  __TBB_Yield();
539  }
540  } else {
541  while( ticket_for_sleep<max_ticket ) {
542  unique_lock<M> ul( my_mtx, defer_lock );
543  ul.lock();
544  // exercise wait part
545  int my_ticket = ++ticket_for_sleep; // grab my ticket
546  if( my_ticket>max_ticket ) break;
547 
548  // each waiter should go to sleep at least once
549  unsigned nw = ++n_waiters;
550  for( ;; ) {
551  // update to max_waitq_length
552  if( nw>max_waitq_length ) max_waitq_length = nw;
553  ++n_visit_to_waitq;
554  test_cv.wait( ul );
555  // if( ret==false ) ++n_timedout;
556  ASSERT( ul, "mutex should have been locked" );
557  --n_waiters;
558  if( signaled_ticket==my_ticket ) {
559  wokeup_ticket = my_ticket;
560  break;
561  }
562  if( n_waiters>0 )
563  test_cv.notify_one();
564  nw = ++n_waiters; // update to max_waitq_length occurs above
565  }
566 
567  ul.unlock();
568  __TBB_Yield(); // give other threads chance to run.
569  }
570  }
571  ++n_done;
572  spin_wait_until_eq( n_done, nthread );
573  ASSERT( n_signaled==max_ticket, "incorrect number of notifications sent" );
574  }
