51 // helper for calculating number of range objects created before balancing phase is started
52 // and for finding maximum and minimum number of iterations among all such ranges
53 // Note: class does not provide exclusive access to members
54 class RangeStatisticCollector {
55 public:
56  RangeStatisticCollector(RangeStatisticData *statisticData) :
57  m_statData(statisticData)
58  {
59  m_called = false;
60  if (m_statData)
61  m_statData->m_rangeNum = 1;
62  }
63 
64  // constructor is called from non-proportional split constructor of derived Range
65  RangeStatisticCollector(RangeStatisticCollector& sc, size_t rangeSize) {
66  if (!sc.m_called) {
67  // this is the first time non-proportional split constructor is called
68  // it means that work distribution phase has been completed and
69  // work balancing phase has been just started
70  sc.m_called = true;
71 
72  if (sc.m_statData) {
73  size_t *minRangeSize = &sc.m_statData->m_minRangeSize;
74  if (*minRangeSize > rangeSize || !sc.m_statData->m_wasMinRangeSizeWritten) { // if minimum is not an actual minimum
75  *minRangeSize = rangeSize;
76  sc.m_statData->m_wasMinRangeSizeWritten = true;
77  }
78  size_t *maxRangeSize = &sc.m_statData->m_maxRangeSize;
79  if (*maxRangeSize < rangeSize) { // if maximum is not an actual maximum
80  *maxRangeSize = rangeSize;
81  }
82  }
83  }
84  *this = sc;
85  // constructor is used on work balancing phase only, so no need to increment
86  // number of range objects created
87  }
88 
89  RangeStatisticCollector(RangeStatisticCollector& sc, proportional_split&) {
90  if (sc.m_statData)
91  sc.m_statData->m_rangeNum++;
92  *this = sc;
93  }
94 
95 private:
96  RangeStatisticData *m_statData;
97 
98  // turns to 'true' when non-proportional split constructor is called first time
99  bool m_called;
100 };

104 template <typename DerivedRange, typename T>
105 class RangeBase: public RangeStatisticCollector {
106 protected:
107  size_t my_begin, my_end;
108  bool m_provide_feedback;
109  bool m_ensure_non_empty_size;
110 public:
111  RangeBase(size_t _begin, size_t _end, RangeStatisticData *statData,
112  bool provide_feedback, bool ensure_non_empty_size)
113  : RangeStatisticCollector(statData)
114  , my_begin(_begin), my_end(_end)
115  , m_provide_feedback(provide_feedback)
116  , m_ensure_non_empty_size(ensure_non_empty_size)
117  { }
118  RangeBase(RangeBase& r, tbb::split) : RangeStatisticCollector(r, r.size()) {
119  *this = r;
120  size_t middle = r.my_begin + (r.my_end - r.my_begin) / 2u;
121  r.my_end = my_begin = middle;
122  }
123 
124  RangeBase(RangeBase& r, proportional_split& p)
125  : RangeStatisticCollector(r, p) {
126  *this = r;
127  size_t original_size = r.size();
128  T right = self().compute_right_part(r, p);
129  size_t right_part = self().round(right);
130  if( m_ensure_non_empty_size ) {
131  right_part = (original_size == right_part) ? (original_size - 1) : right_part;
132  right_part = (right_part != 0) ? right_part : 1;
133  }
134  r.my_end = my_begin = r.my_end - right_part;
135 #if __TBB_ENABLE_RANGE_FEEDBACK
136  if( m_provide_feedback )
137  p.set_proportion(original_size - right_part, right_part);
138 #endif
139  if( m_ensure_non_empty_size )
140  ASSERT(r.my_end != r.my_begin && my_end != my_begin, "Incorrect range split");
141  }
142 
143  size_t begin() const { return my_begin; }
144  size_t end() const { return my_end; }
145  bool is_divisible() const { return (my_end - my_begin) > 1; }
146  bool empty() const { return my_end == my_begin; }
147  size_t size() const { return my_end - my_begin; }
148 
149  // helper methods (not part of the range concept)
150  DerivedRange& self() { return static_cast<DerivedRange&>(*this); }
151  size_t round(T part) { return size_t(part); }
152  T compute_right_part(RangeBase& r, proportional_split& p) {
153  return T(r.size() * T(p.right())) / T(p.left() + p.right());
154  }
155  bool is_ensure_non_emptiness() { return m_ensure_non_empty_size; }
156 };

281  : RangeBase<ExactSplitRange, size_t>(r, p) { }
282  size_t compute_right_part(RangeBase<ExactSplitRange, size_t>& r, proportional_split& p) {
283  size_t parts = size_t(p.left() + p.right());
284  size_t currSize = r.size();
285  size_t int_part = currSize / parts * p.right();
286  size_t remainder = currSize % parts * p.right();
287  int_part += remainder / parts;
288  remainder %= parts;
289  size_t right_part = int_part + (remainder > parts/2 ? 1 : 0);
290  return right_part;
291  }
292  static const bool is_splittable_in_proportion = true;
293 };

353  void push_subnode(TreeNode *&root_node, TreeNode *node) {
354  if (!root_node) {
355  root_node = node;
356  return;
357  } else if (are_nodes_equal(root_node, node)) {
358  // no need to push the same node
359  return;
360  }
361 
362  if (!has_children(root_node)) {
363  // if current root_node does not have children passed node
364  // should has one of the interval bounds to be equal to
365  // the same bound in the root_node
366  if (is_look_like_left_sibling(root_node, node))
367  push_subnode(root_node->m_left, node);
368  else
369  push_subnode(root_node->m_right, node);
370  return;
371  }
372 
373  if (has_left_child(root_node)) {
374  if (is_subnode(root_node->m_left, node)) {
375  push_subnode(root_node->m_left, node);
376  return;
377  }
378  push_subnode(root_node->m_right, node);
379  return;
380  }
381 
382  ASSERT(root_node->m_right != NULL, "Right child is NULL but must be present");
383  if (is_subnode(root_node->m_right, node)) {
384  push_subnode(root_node->m_right, node);
385  return;
386  }
387  push_subnode(root_node->m_left, node);
388  return;
389  }
390 
391  bool has_children(TreeNode *node) { return node->m_left || node->m_right; }
392 
393  bool is_look_like_left_sibling(TreeNode *root_node, TreeNode *node) {
394  if (root_node->m_range_begin == node->m_range_begin)
395  return true;
396  ASSERT(root_node->m_range_end == node->m_range_end, NULL);
397  return false;
398  }
399 
400  bool has_left_child(TreeNode *node) { return node->m_left != NULL; }
401 
402  bool is_subnode(TreeNode *root_node, TreeNode *node) {
403  return root_node->m_range_begin <= node->m_range_begin &&
404  node->m_range_end <= root_node->m_range_end;
405  }
406 
407  bool are_nodes_equal(TreeNode *node1, TreeNode *node2) const {

490  * Possible use cases are:
491  * -------------------------------------------------------------------------------------------------------------
492  * Range# is_splittable_in_proportion Range proportional ctor Used partitioner Result Effect
493  * -------------------------------------------------------------------------------------------------------------
494  * 1 true available proportional pMN, r(p), part(p)
495  * -------------------------------------------------------------------------------------------------------------
496  * 2 false available proportional p11, r(p), part(p)
497  * -------------------------------------------------------------------------------------------------------------
498  * 3 not defined available proportional p11, r(p), part(p)
499  * -------------------------------------------------------------------------------------------------------------
500  * 4 true not available proportional pMN, r(s), part(p) *
501  * -------------------------------------------------------------------------------------------------------------
502  * 5 false not available proportional p11, r(s), part(p)
503  * -------------------------------------------------------------------------------------------------------------
504  * 6 not defined not available proportional p11, r(s), part(p)
505  * -------------------------------------------------------------------------------------------------------------
506  * 1 true available simple s, r(s), part(s)
507  * -------------------------------------------------------------------------------------------------------------
508  * 2 false available simple s, r(s), part(s)
509  * -------------------------------------------------------------------------------------------------------------
510  * 3 not defined available simple s, r(s), part(s)
511  * -------------------------------------------------------------------------------------------------------------
512  * 4 true not available simple s, r(s), part(s)
513  * -------------------------------------------------------------------------------------------------------------
514  * 5 false not available simple s, r(s), part(s)
515  * -------------------------------------------------------------------------------------------------------------
516  * 6 not defined not available simple s, r(s), part(s)
517  * -------------------------------------------------------------------------------------------------------------
518  *
519  * Legend:
520  * proportional - with proportional splits (e.g. affinity_partitioner)
521  * simple - without proportional splits (e.g. simple_partitioner, auto_partitioner)
522  * pMN - proportional_split object with proportion M to N is created. (p11 - proportion 1 to 1)
523  * s - split object is created
524  * r(p) - range's proportional split constructor is called
525  * r(s) - range's ordinary split constructor is called
526  * part(p) - partitioner's proportional split constructor is called
527  * part(s) - partitioner's ordinary split constructor is called
528  * * - incorrect split behavior is possible (e.g. partitioner divides at an arbitrary ratio while
529  * range divides into halves)
