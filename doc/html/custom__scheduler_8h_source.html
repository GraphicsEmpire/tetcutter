90  void tally_completion_of_predecessor( task& s, task*& bypass_slot ) {
91  task_prefix& p = s.prefix();
92  if( SchedulerTraits::itt_possible )
93  ITT_NOTIFY(sync_releasing, &p.ref_count);
94  if( SchedulerTraits::has_slow_atomic && p.ref_count==1 )
95  p.ref_count=0;
96  else if( __TBB_FetchAndDecrementWrelease(&p.ref_count) > 1 ) {// more references exist
97  // '__TBB_cl_evict(&p)' degraded performance of parallel_preorder example
98  return;
99  }
100 
101  // Ordering on p.ref_count (superfluous if SchedulerTraits::has_slow_atomic)
102  __TBB_control_consistency_helper();
103  __TBB_ASSERT(p.ref_count==0, "completion of task caused predecessor's reference count to underflow");
104  if( SchedulerTraits::itt_possible )
105  ITT_NOTIFY(sync_acquired, &p.ref_count);
106 #if TBB_USE_ASSERT
107  p.extra_state &= ~es_ref_count_active;
108 #endif /* TBB_USE_ASSERT */
109 
110 #if __TBB_RECYCLE_TO_ENQUEUE
111  if (p.state==task::to_enqueue) {
112  // related to __TBB_TASK_ARENA TODO: try keep priority of the task
113  // e.g. rework task_prefix to remember priority of received task and use here
114  my_arena->enqueue_task(s, 0, my_random );
115  } else
116 #endif /*__TBB_RECYCLE_TO_ENQUEUE*/
117  if( bypass_slot==NULL )
118  bypass_slot = &s;
119  else
120  local_spawn( s, s.prefix().next );
121  }

141 template<typename SchedulerTraits>
142 task* custom_scheduler<SchedulerTraits>::receive_or_steal_task( __TBB_atomic reference_count& completion_ref_count ) {
143  task* t = NULL;
144  bool outermost_worker_level = worker_outermost_level();
145  bool outermost_dispatch_level = outermost_worker_level || master_outermost_level();
146  bool can_steal_here = can_steal();
147  my_inbox.set_is_idle( true );
148 #if __TBB_HOARD_NONLOCAL_TASKS
149  __TBB_ASSERT(!my_nonlocal_free_list, NULL);
150 #endif
151 #if __TBB_TASK_PRIORITY
152  if ( outermost_dispatch_level ) {
153  if ( intptr_t skipped_priority = my_arena->my_skipped_fifo_priority ) {
154  // This thread can dequeue FIFO tasks, and some priority levels of
155  // FIFO tasks have been bypassed (to prevent deadlock caused by
156  // dynamic priority changes in nested task group hierarchy).
157  if ( my_arena->my_skipped_fifo_priority.compare_and_swap(0, skipped_priority) == skipped_priority
158  && skipped_priority > my_arena->my_top_priority )
159  {
160  my_market->update_arena_priority( *my_arena, skipped_priority );
161  }
162  }
163  }
164 #endif /* !__TBB_TASK_PRIORITY */
165  // TODO: Try to find a place to reset my_limit (under market's lock)
166  // The number of slots potentially used in the arena. Updated once in a while, as my_limit changes rarely.
167  size_t n = my_arena->my_limit-1;
168  int yield_count = 0;
169  // The state "failure_count==-1" is used only when itt_possible is true,
170  // and denotes that a sync_prepare has not yet been issued.
171  for( int failure_count = -static_cast<int>(SchedulerTraits::itt_possible);; ++failure_count) {
172  __TBB_ASSERT( my_arena->my_limit > 0, NULL );
173  __TBB_ASSERT( my_arena_index <= n, NULL );
174  if( completion_ref_count==1 ) {
175  if( SchedulerTraits::itt_possible ) {
176  if( failure_count!=-1 ) {
177  ITT_NOTIFY(sync_prepare, &completion_ref_count);
178  // Notify Intel(R) Thread Profiler that thread has stopped spinning.
179  ITT_NOTIFY(sync_acquired, this);
180  }
181  ITT_NOTIFY(sync_acquired, &completion_ref_count);
182  }
183  __TBB_ASSERT( !t, NULL );
184  __TBB_control_consistency_helper(); // on ref_count
185  break; // exit stealing loop and return;
186  }
187  // Check if the resource manager requires our arena to relinquish some threads
188  if ( outermost_worker_level && my_arena->my_num_workers_allotted < my_arena->num_workers_active() ) {
189 #if !__TBB_TASK_ARENA
190  __TBB_ASSERT( is_worker(), NULL );
191 #endif
192  if( SchedulerTraits::itt_possible && failure_count != -1 )
193  ITT_NOTIFY(sync_cancel, this);
194  return NULL;
195  }
196 #if __TBB_TASK_PRIORITY
197  const int p = int(my_arena->my_top_priority);
198 #else /* !__TBB_TASK_PRIORITY */
199  static const int p = 0;
200 #endif
201  // Check if there are tasks mailed to this thread via task-to-thread affinity mechanism.
202  __TBB_ASSERT(my_affinity_id, NULL);
203  if ( n && !my_inbox.empty() && (t = get_mailbox_task()) ) {
204  GATHER_STATISTIC( ++my_counters.mails_received );
205  }
206  // Check if there are tasks in starvation-resistant stream.
207  // Only allowed at the outermost dispatch level.
208  else if ( outermost_dispatch_level && !my_arena->my_task_stream.empty(p)
209  && (t = my_arena->my_task_stream.pop( p, my_arena_slot->hint_for_pop)) ) {
210  ITT_NOTIFY(sync_acquired, &my_arena->my_task_stream);
211  // just proceed with the obtained task
212  }
213 #if __TBB_TASK_PRIORITY
214  // Check if any earlier offloaded non-top priority tasks become returned to the top level
215  else if ( my_offloaded_tasks && (t=reload_tasks()) ) {
216  // just proceed with the obtained task
217  }
218 #endif /* __TBB_TASK_PRIORITY */
219  else if ( can_steal_here && n ) {
220  // Try to steal a task from a random victim.
221  size_t k = my_random.get() % n;
222  arena_slot* victim = &my_arena->my_slots[k];
223  // The following condition excludes the master that might have
224  // already taken our previous place in the arena from the list .
225  // of potential victims. But since such a situation can take
226  // place only in case of significant oversubscription, keeping
227  // the checks simple seems to be preferable to complicating the code.
228  if( k >= my_arena_index )
229  ++victim; // Adjusts random distribution to exclude self
230  task **pool = victim->task_pool;
231  if( pool == EmptyTaskPool || !(t = steal_task( *victim )) )
232  goto fail;
233  if( is_proxy(*t) ) {
234  task_proxy &tp = *(task_proxy*)t;
235  t = tp.extract_task<task_proxy::pool_bit>();
236  if ( !t ) {
237  // Proxy was empty, so it's our responsibility to free it
238  free_task<no_cache_small_task>(tp);
239  goto fail;
240  }
241  GATHER_STATISTIC( ++my_counters.proxies_stolen );
242  }
243  t->prefix().extra_state |= es_task_is_stolen;
244  if( is_version_3_task(*t) ) {
245  my_innermost_running_task = t;
246  t->prefix().owner = this;
247  t->note_affinity( my_affinity_id );
248  }
249  GATHER_STATISTIC( ++my_counters.steals_committed );
250  } // end of stealing branch
251  else
252  goto fail;
253  // A task was successfully obtained somewhere
254  __TBB_ASSERT(t,NULL);
255 #if __TBB_SCHEDULER_OBSERVER
256  my_arena->my_observers.notify_entry_observers( my_last_local_observer, is_worker() );
257  the_global_observer_list.notify_entry_observers( my_last_global_observer, is_worker() );
258 #endif /* __TBB_SCHEDULER_OBSERVER */
259  if ( SchedulerTraits::itt_possible && failure_count != -1 ) {
260  // FIXME - might be victim, or might be selected from a mailbox
261  // Notify Intel(R) Thread Profiler that thread has stopped spinning.
262  ITT_NOTIFY(sync_acquired, this);
263  }
264  break; // exit stealing loop and return
265 fail:
266  GATHER_STATISTIC( ++my_counters.steals_failed );
267  if( SchedulerTraits::itt_possible && failure_count==-1 ) {
268  // The first attempt to steal work failed, so notify Intel(R) Thread Profiler that
269  // the thread has started spinning. Ideally, we would do this notification
270  // *before* the first failed attempt to steal, but at that point we do not
271  // know that the steal will fail.
272  ITT_NOTIFY(sync_prepare, this);
273  failure_count = 0;
274  }
275  // Pause, even if we are going to yield, because the yield might return immediately.
276  __TBB_Pause(PauseTime);
277  const int failure_threshold = 2*int(n+1);
278  if( failure_count>=failure_threshold ) {
279 #if __TBB_YIELD2P
280  failure_count = 0;
281 #else
282  failure_count = failure_threshold;
283 #endif
284  __TBB_Yield();
285 #if __TBB_TASK_PRIORITY
286  // Check if there are tasks abandoned by other workers
287  if ( my_arena->my_orphaned_tasks ) {
288  // Epoch must be advanced before seizing the list pointer
289  ++my_arena->my_abandonment_epoch;
290  task* orphans = (task*)__TBB_FetchAndStoreW( &my_arena->my_orphaned_tasks, 0 );
291  if ( orphans ) {
292  task** link = NULL;
293  // Get local counter out of the way (we've just brought in external tasks)
294  my_local_reload_epoch--;
295  t = reload_tasks( orphans, link, effective_reference_priority() );
296  if ( orphans ) {
297  *link = my_offloaded_tasks;
298  if ( !my_offloaded_tasks )
299  my_offloaded_task_list_tail_link = link;
300  my_offloaded_tasks = orphans;
301  }
302  __TBB_ASSERT( !my_offloaded_tasks == !my_offloaded_task_list_tail_link, NULL );
303  if ( t ) {
304  if( SchedulerTraits::itt_possible )
305  ITT_NOTIFY(sync_cancel, this);
306  break; // exit stealing loop and return
307  }
308  }
309  }
310 #endif /* __TBB_TASK_PRIORITY */
311  const int yield_threshold = 100;
312  if( yield_count++ >= yield_threshold ) {
313  // When a worker thread has nothing to do, return it to RML.
314  // For purposes of affinity support, the thread is considered idle while in RML.
315 #if __TBB_TASK_PRIORITY
316  if( outermost_worker_level || my_arena->my_top_priority > my_arena->my_bottom_priority ) {
317  if ( my_arena->is_out_of_work() && outermost_worker_level ) {
318 #else /* !__TBB_TASK_PRIORITY */
319  if ( outermost_worker_level && my_arena->is_out_of_work() ) {
320 #endif /* !__TBB_TASK_PRIORITY */
321  if( SchedulerTraits::itt_possible )
322  ITT_NOTIFY(sync_cancel, this);
323  return NULL;
324  }
325 #if __TBB_TASK_PRIORITY
326  }
327  if ( my_offloaded_tasks ) {
328  // Safeguard against any sloppiness in managing reload epoch
329  // counter (e.g. on the hot path because of performance reasons).
330  my_local_reload_epoch--;
331  // Break the deadlock caused by a higher priority dispatch loop
332  // stealing and offloading a lower priority task. Priority check
333  // at the stealing moment cannot completely preclude such cases
334  // because priorities can changes dynamically.
335  if ( !outermost_worker_level && *my_ref_top_priority > my_arena->my_top_priority ) {
336  GATHER_STATISTIC( ++my_counters.prio_ref_fixups );
337  my_ref_top_priority = &my_arena->my_top_priority;
338  // it's expected that only outermost workers can use global reload epoch
339  __TBB_ASSERT(my_ref_reload_epoch == &my_arena->my_reload_epoch, NULL);
340  }
341  }
342 #endif /* __TBB_TASK_PRIORITY */
343  } // end of arena snapshot branch
344  // If several attempts did not find work, re-read the arena limit.
345  n = my_arena->my_limit-1;
346  } // end of yielding branch
347  } // end of nonlocal task retrieval loop
348  my_inbox.set_is_idle( false );
349  return t;
350 }

352 template<typename SchedulerTraits>
353 void custom_scheduler<SchedulerTraits>::local_wait_for_all( task& parent, task* child ) {
354  __TBB_ASSERT( governor::is_set(this), NULL );
355  __TBB_ASSERT( parent.ref_count() >= (child && child->parent() == &parent ? 2 : 1), "ref_count is too small" );
356  assert_task_pool_valid();
357  // Using parent's refcount in sync_prepare (in the stealing loop below) is
358  // a workaround for TP. We need to name it here to display correctly in Ampl.
359  if( SchedulerTraits::itt_possible )
360  ITT_SYNC_CREATE(&parent.prefix().ref_count, SyncType_Scheduler, SyncObj_TaskStealingLoop);
361 #if __TBB_TASK_GROUP_CONTEXT
362  __TBB_ASSERT( parent.prefix().context || (is_worker() && &parent == my_dummy_task), "parent task does not have context" );
363 #endif /* __TBB_TASK_GROUP_CONTEXT */
364  task* t = child;
365  // Constant all_local_work_done is an unreachable refcount value that prevents
366  // early quitting the dispatch loop. It is defined to be in the middle of the range
367  // of negative values representable by the reference_count type.
368  static const reference_count
369  // For normal dispatch loops
370  parents_work_done = 1,
371  // For termination dispatch loops in masters
372  all_local_work_done = (reference_count)3 << (sizeof(reference_count) * 8 - 2);
373  reference_count quit_point;
374 #if __TBB_TASK_PRIORITY
375  __TBB_ASSERT( (uintptr_t)*my_ref_top_priority < (uintptr_t)num_priority_levels, NULL );
376  volatile intptr_t *old_ref_top_priority = my_ref_top_priority;
377  // When entering nested parallelism level market level counter
378  // must be replaced with the one local to this arena.
379  volatile uintptr_t *old_ref_reload_epoch = my_ref_reload_epoch;
380 #endif /* __TBB_TASK_PRIORITY */
381  task* old_dispatching_task = my_dispatching_task;
382  my_dispatching_task = my_innermost_running_task;
383  if( master_outermost_level() ) {
384  // We are in the outermost task dispatch loop of a master thread or a worker which mimics master
385  __TBB_ASSERT( !is_worker() || my_dispatching_task != old_dispatching_task, NULL );
386  quit_point = &parent == my_dummy_task ? all_local_work_done : parents_work_done;
387  } else {
388  quit_point = parents_work_done;
389 #if __TBB_TASK_PRIORITY
390  if ( &parent != my_dummy_task ) {
391  // We are in a nested dispatch loop.
392  // Market or arena priority must not prevent child tasks from being
393  // executed so that dynamic priority changes did not cause deadlock.
394  my_ref_top_priority = &parent.prefix().context->my_priority;
395  my_ref_reload_epoch = &my_arena->my_reload_epoch;
396  if(my_ref_reload_epoch != old_ref_reload_epoch)
397  my_local_reload_epoch = *my_ref_reload_epoch-1;
398  }
399 #endif /* __TBB_TASK_PRIORITY */
400  }
401 
402  cpu_ctl_env_helper cpu_ctl_helper;
403  if ( t )
404  cpu_ctl_helper.set_env( __TBB_CONTEXT_ARG1(t->prefix().context) );
405 
406 #if TBB_USE_EXCEPTIONS
407  // Infinite safeguard EH loop
408  for (;;) {
409  try {
410 #endif /* TBB_USE_EXCEPTIONS */
411  // Outer loop receives tasks from global environment (via mailbox, FIFO queue(s),
412  // and by stealing from other threads' task pools).
413  // All exit points from the dispatch loop are located in its immediate scope.
414  for(;;) {
415  // Middle loop retrieves tasks from the local task pool.
416  for(;;) {
417  // Inner loop evaluates tasks coming from nesting loops and those returned
418  // by just executed tasks (bypassing spawn or enqueue calls).
419  while(t) {
420  __TBB_ASSERT( my_inbox.is_idle_state(false), NULL );
421  __TBB_ASSERT(!is_proxy(*t),"unexpected proxy");
422  __TBB_ASSERT( t->prefix().owner, NULL );
423  assert_task_valid(*t);
424 #if __TBB_TASK_GROUP_CONTEXT && TBB_USE_ASSERT
425  assert_context_valid(t->prefix().context);
426  if ( !t->prefix().context->my_cancellation_requested )
427 #endif
428  __TBB_ASSERT( 1L<<t->state() & (1L<<task::allocated|1L<<task::ready|1L<<task::reexecute), NULL );
429  assert_task_pool_valid();
430 #if __TBB_TASK_PRIORITY
431  intptr_t p = priority(*t);
432  if ( p != *my_ref_top_priority && (t->prefix().extra_state & es_task_enqueued) == 0) {
433  assert_priority_valid(p);
434  if ( p != my_arena->my_top_priority ) {
435  my_market->update_arena_priority( *my_arena, p );
436  }
437  if ( p < effective_reference_priority() ) {
438  if ( !my_offloaded_tasks ) {
439  my_offloaded_task_list_tail_link = &t->prefix().next_offloaded;
440  // Erase possible reference to the owner scheduler (next_offloaded is a union member)
441  *my_offloaded_task_list_tail_link = NULL;
442  }
443  offload_task( *t, p );
444  if ( in_arena() ) {
445  t = winnow_task_pool();
446  if ( t )
447  continue;
448  }
449  else {
450  // Mark arena as full to unlock arena priority level adjustment
451  // by arena::is_out_of_work(), and ensure worker's presence.
452  my_arena->advertise_new_work<false>();
453  }
454  goto stealing_ground;
455  }
456  }
457 #endif /* __TBB_TASK_PRIORITY */
458  task* t_next = NULL;
459  my_innermost_running_task = t;
460  t->prefix().owner = this;
461  t->prefix().state = task::executing;
462 #if __TBB_TASK_GROUP_CONTEXT
463  if ( !t->prefix().context->my_cancellation_requested )
464 #endif
465  {
466  GATHER_STATISTIC( ++my_counters.tasks_executed );
467  GATHER_STATISTIC( my_counters.avg_arena_concurrency += my_arena->num_workers_active() );
468  GATHER_STATISTIC( my_counters.avg_assigned_workers += my_arena->my_num_workers_allotted );
469 #if __TBB_TASK_PRIORITY
470  GATHER_STATISTIC( my_counters.avg_arena_prio += p );
471  GATHER_STATISTIC( my_counters.avg_market_prio += my_market->my_global_top_priority );
472 #endif /* __TBB_TASK_PRIORITY */
473  ITT_STACK(SchedulerTraits::itt_possible, callee_enter, t->prefix().context->itt_caller);
474  t_next = t->execute();
475  ITT_STACK(SchedulerTraits::itt_possible, callee_leave, t->prefix().context->itt_caller);
476  if (t_next) {
477  __TBB_ASSERT( t_next->state()==task::allocated,
478  "if task::execute() returns task, it must be marked as allocated" );
479  reset_extra_state(t_next);
480 #if TBB_USE_ASSERT
481  affinity_id next_affinity=t_next->prefix().affinity;
482  if (next_affinity != 0 && next_affinity != my_affinity_id)
483  GATHER_STATISTIC( ++my_counters.affinity_ignored );
484 #endif
485  }
486  }
487  assert_task_pool_valid();
488  switch( t->state() ) {
489  case task::executing: {
490  task* s = t->parent();
491  __TBB_ASSERT( my_innermost_running_task==t, NULL );
492  __TBB_ASSERT( t->prefix().ref_count==0, "Task still has children after it has been executed" );
493  t->~task();
494  if( s )
495  tally_completion_of_predecessor(*s, t_next);
496  free_task<no_hint>( *t );
497  assert_task_pool_valid();
498  break;
499  }
500 
501  case task::recycle: // set by recycle_as_safe_continuation()
502  t->prefix().state = task::allocated;
503 #if __TBB_RECYCLE_TO_ENQUEUE
504  case task::to_enqueue: // set by recycle_to_enqueue()
505 #endif
506  __TBB_ASSERT( t_next != t, "a task returned from method execute() can not be recycled in another way" );
507  reset_extra_state(t);
508  // for safe continuation, need atomically decrement ref_count;
509  tally_completion_of_predecessor(*t, t_next);
510  assert_task_pool_valid();
511  break;
512 
513  case task::reexecute: // set by recycle_to_reexecute()
514  __TBB_ASSERT( t_next, "reexecution requires that method execute() return another task" );
515  __TBB_ASSERT( t_next != t, "a task returned from method execute() can not be recycled in another way" );
516  t->prefix().state = task::allocated;
517  reset_extra_state(t);
518  local_spawn( *t, t->prefix().next );
519  assert_task_pool_valid();
520  break;
521  case task::allocated:
522  reset_extra_state(t);
523  break;
524 #if TBB_USE_ASSERT
525  case task::ready:
526  __TBB_ASSERT( false, "task is in READY state upon return from method execute()" );
